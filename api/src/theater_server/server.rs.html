203
204// Allow converting from TheaterRuntimeError to ManagementError
205impl From<TheaterRuntimeError> for ManagementError {
206    fn from(err: TheaterRuntimeError) -> Self {
207        match err {
208            TheaterRuntimeError::ActorNotFound(_) => ManagementError::ActorNotFound,
209            TheaterRuntimeError::ActorAlreadyExists(_) => ManagementError::ActorAlreadyExists,
210            TheaterRuntimeError::ActorNotRunning(_) => ManagementError::ActorNotRunning,
211            TheaterRuntimeError::ActorOperationFailed(msg) => {
212                ManagementError::RuntimeError(format!("Actor operation failed: {}", msg))
213            }
214            TheaterRuntimeError::ActorError(e) => ManagementError::ActorError(e.to_string()),
215            TheaterRuntimeError::ChannelError(msg) => ManagementError::CommunicationError(msg),
216            TheaterRuntimeError::ChannelNotFound(_) => ManagementError::ChannelNotFound,
217            TheaterRuntimeError::ChannelRejected => ManagementError::ChannelRejected,
218            TheaterRuntimeError::SerializationError(msg) => {
219                ManagementError::SerializationError(msg)
220            }
221            TheaterRuntimeError::InternalError(msg) => ManagementError::InternalError(msg),
222        }
223    }

257pub struct TheaterServer {

259    theater_tx: mpsc::Sender<TheaterCommand>,

262    // Field to track channel subscriptions
263    channel_subscriptions: Arc<Mutex<HashMap<String, ChannelSubscription>>>,
264    // Channel for runtime to send channel events back to server
265    #[allow(dead_code)]
266    channel_events_tx: mpsc::Sender<ChannelEvent>,

269impl TheaterServer {
270    // Process channel events and forward them to subscribed clients
271    async fn process_channel_events(
272        mut channel_events_rx: mpsc::Receiver<ChannelEvent>,
273        channel_subscriptions: Arc<Mutex<HashMap<String, ChannelSubscription>>>,
274    ) {
275        while let Some(event) = channel_events_rx.recv().await {
276            match event {
277                ChannelEvent::Message {
278                    channel_id,
279                    sender_id,
280                    message,
281                } => {
282                    tracing::debug!("Received channel message for {}", channel_id);
283                    // Forward to subscribed clients
284                    let subs = channel_subscriptions.lock().await;
285                    if let Some(sub) = subs.get(&channel_id.0) {
286                        let response = ManagementResponse::ChannelMessage {
287                            channel_id: channel_id.0.clone(),
288                            sender_id,
289                            message,
290                        };
291
292                        if let Err(e) = sub.client_tx.send(response).await {
293                            tracing::warn!("Failed to forward channel message: {}", e);
294                        }
295                    }
296                }
297                ChannelEvent::Close { channel_id } => {
298                    tracing::debug!("Received channel close event for {}", channel_id);
299                    // Forward to subscribed clients
300                    let mut subs = channel_subscriptions.lock().await;
301                    if let Some(sub) = subs.remove(&channel_id.0) {
302                        let response = ManagementResponse::ChannelClosed {
303                            channel_id: channel_id.0.clone(),
304                        };
305
306                        if let Err(e) = sub.client_tx.send(response).await {
307                            tracing::warn!("Failed to forward channel close event: {}", e);
308                        }
309                    }
310                }
311            }
312        }
313    }

315    pub async fn new(address: std::net::SocketAddr) -> Result<Self> {
316        let (theater_tx, theater_rx) = mpsc::channel(32);
317
318        // Create channel for runtime to send channel events back to server
319        let (channel_events_tx, channel_events_rx) = mpsc::channel(32);
320
321        // Pass channel_events_tx to runtime during initialization
322        let runtime = TheaterRuntime::new(
323            theater_tx.clone(),
324            theater_rx,
325            Some(channel_events_tx.clone()),
326        )
327        .await?;
328        let management_socket = TcpListener::bind(address).await?;
329
330        let channel_subscriptions = Arc::new(Mutex::new(HashMap::new()));
331
332        // Start task to process channel events
333        let channel_subs_clone = channel_subscriptions.clone();
334        tokio::spawn(async move {
335            Self::process_channel_events(channel_events_rx, channel_subs_clone).await;
336        });
337
338        Ok(Self {
339            runtime,
340            theater_tx,
341            management_socket,
342            subscriptions: Arc::new(Mutex::new(HashMap::new())),
343            channel_subscriptions,
344            channel_events_tx,
345        })
346    }

348    pub async fn run(mut self) -> Result<()> {
349        info!(
350            "Theater server starting on {:?}",
351            self.management_socket.local_addr()?
352        );
353
354        // Start the theater runtime in its own task
355        let runtime_handle = tokio::spawn(async move {
356            match self.runtime.run().await {
357                Ok(_) => Ok(()),
358                Err(e) => {
359                    error!("Theater runtime failed: {}", e);
360                    Err(e)
361                }
362            }
363        });
364
365        // Accept and handle management connections
366        while let Ok((socket, addr)) = self.management_socket.accept().await {
367            info!("New management connection from {}", addr);
368            let runtime_tx = self.theater_tx.clone();
369            let subscriptions = self.subscriptions.clone();
370            let channel_subscriptions = self.channel_subscriptions.clone();
371
372            tokio::spawn(async move {
373                if let Err(e) = Self::handle_management_connection(
374                    socket,
375                    runtime_tx,
376                    subscriptions,
377                    channel_subscriptions,
378                )
379                .await
380                {
381                    error!("Error handling management connection: {}", e);
382                }
383            });
384        }
385
386        runtime_handle.await??;
387        Ok(())
388    }

390    async fn handle_management_connection(
391        socket: TcpStream,
392        runtime_tx: mpsc::Sender<TheaterCommand>,
393        subscriptions: Arc<Mutex<HashMap<TheaterId, HashSet<Subscription>>>>,
394        channel_subscriptions: Arc<Mutex<HashMap<String, ChannelSubscription>>>,
395    ) -> Result<()> {
396        // Create a channel for sending responses to this client
397        let (client_tx, mut client_rx) = mpsc::channel::<ManagementResponse>(32);
398
399        let mut codec = LengthDelimitedCodec::new();
400        codec.set_max_frame_length(32 * 1024 * 1024); // Increase to 32MB
401        let framed = Framed::new(socket, codec);
402
403        // Split the framed connection into read and write parts
404        let (mut framed_sink, mut framed_stream) = framed.split();
405
406        // Clone the client_tx for use in the command loop
407        let cmd_client_tx = client_tx.clone();
408
409        // Start a task to forward responses to the client
410        let _response_task = tokio::spawn(async move {
411            while let Some(response) = client_rx.recv().await {
412                match serde_json::to_vec(&response) {
413                    Ok(data) => {
414                        if let Err(e) = framed_sink.send(Bytes::from(data)).await {
415                            debug!("Error sending response to client: {}", e);
416                            break;
417                        }
418                    }
419                    Err(e) => {
420                        error!("Error serializing response: {}", e);
421                    }
422                }
423            }
424            debug!("Response forwarder for client closed");
425        });
426
427        // Store active subscriptions for this connection to clean up on disconnect
428        let mut connection_subscriptions: Vec<(TheaterId, Uuid)> = Vec::new();
429
430        // Store active channel subscriptions for cleanup
431        let mut connection_channel_subscriptions: Vec<String> = Vec::new();
432
433        // Loop until connection closes or an error occurs
434        'connection: while let Some(msg) = framed_stream.next().await {
435            debug!("Received management message");
436            let msg = match msg {
437                Ok(m) => m,
438                Err(e) => {
439                    error!("Error receiving message: {}", e);
440                    break 'connection;
441                }
442            };
443
444            let cmd = match serde_json::from_slice::<ManagementCommand>(&msg) {
445                Ok(c) => c,
446                Err(e) => {
447                    error!(
448                        "Error parsing command: {} {}",
449                        e,
450                        String::from_utf8_lossy(&msg)
451                    );
452                    continue;
453                }
454            };
455            debug!("Parsed command: {:?}", cmd);
456
457            // Store the command for reference (used for subscription tracking)
458            let _cmd_clone = cmd.clone();
459
460            let response = match cmd {
461                ManagementCommand::StartActor {
462                    manifest,
463                    initial_state,
464                    parent,
465                    subscribe,
466                } => {
467                    info!("Starting actor from manifest: {:?}", manifest);
468                    let (cmd_tx, cmd_rx) = tokio::sync::oneshot::channel();
469                    debug!("Sending SpawnActor command to runtime");
470                    let supervisor_tx = if parent {
471                        let (supervisor_tx, mut supervisor_rx) = mpsc::channel(32);
472                        let cmd_client_tx = cmd_client_tx.clone();
473                        tokio::spawn(async move {
474                            while let Some(res) = supervisor_rx.recv().await {
475                                debug!("Received supervisor response: {:?}", res);
476                                if let Err(e) = cmd_client_tx
477                                    .send(ManagementResponse::ActorResult(res))
478                                    .await
479                                {
480                                    error!("Failed to send supervisor response: {}", e);
481                                    break;
482                                }
483                            }
484                        });
485                        Some(supervisor_tx)
486                    } else {
487                        None
488                    };
489                    let subscription_tx = if subscribe {
490                        let (event_tx, mut event_rx) = mpsc::channel(32);
491
492                        // set up a task to forward events to the client
493                        let cmd_client_tx = cmd_client_tx.clone();
494                        tokio::spawn(async move {
495                            while let Some(event) = event_rx.recv().await {
496                                debug!("Received event for subscription");
497                                let response = match event {
498                                    Ok(event) => ManagementResponse::ActorEvent { event },
499                                    Err(e) => ManagementResponse::ActorError { error: e },
500                                };
501                                if let Err(e) = cmd_client_tx.send(response).await {
502                                    debug!("Failed to forward event to client: {}", e);
503                                    break;
504                                }
505                            }
506                            debug!("Event forwarder for subscription stopped");
507                        });
508
509                        Some(event_tx)
510                    } else {
511                        None
512                    };
513                    match runtime_tx
514                        .send(TheaterCommand::SpawnActor {
515                            manifest_path: manifest.clone(),
516                            init_bytes: initial_state,
517                            response_tx: cmd_tx,
518                            parent_id: None,
519                            supervisor_tx,
520                            subscription_tx,
521                        })
522                        .await
523                    {
524                        Ok(_) => {
525                            debug!("SpawnActor command sent to runtime, awaiting response");
526                            match cmd_rx.await {
527                                Ok(result) => match result {
528                                    Ok(actor_id) => {
529                                        info!("Actor started with ID: {:?}", actor_id);
530                                        ManagementResponse::ActorStarted { id: actor_id }
531                                    }
532                                    Err(e) => {
533                                        error!("Runtime failed to start actor: {}", e);
534                                        ManagementResponse::Error {
535                                            error: ManagementError::RuntimeError(format!(
536                                                "Failed to start actor: {}",
537                                                e
538                                            )),
539                                        }
540                                    }
541                                },
542                                Err(e) => {
543                                    error!("Failed to receive spawn response: {}", e);
544                                    ManagementResponse::Error {
545                                        error: ManagementError::CommunicationError(format!(
546                                            "Failed to receive spawn response: {}",
547                                            e
548                                        )),
549                                    }
550                                }
551                            }
552                        }
553                        Err(e) => {
554                            error!("Failed to send SpawnActor command: {}", e);
555                            ManagementResponse::Error {
556                                error: ManagementError::CommunicationError(format!(
557                                    "Failed to send spawn command: {}",
558                                    e
559                                )),
560                            }
561                        }
562                    }
563                }
564                ManagementCommand::StopActor { id } => {
565                    info!("Stopping actor: {:?}", id);
566                    let (cmd_tx, cmd_rx) = tokio::sync::oneshot::channel();
567                    runtime_tx
568                        .send(TheaterCommand::StopActor {
569                            actor_id: id.clone(),
570                            response_tx: cmd_tx,
571                        })
572                        .await?;
573
574                    match cmd_rx.await? {
575                        Ok(_) => {
576                            subscriptions.lock().await.remove(&id);
577                            ManagementResponse::ActorStopped { id }
578                        }
579                        Err(e) => ManagementResponse::Error {
580                            error: ManagementError::RuntimeError(format!(
581                                "Failed to stop actor: {}",
582                                e
583                            )),
584                        },
585                    }
586                }
587                ManagementCommand::ListActors => {
588                    debug!("Listing actors");
589                    let (cmd_tx, cmd_rx) = tokio::sync::oneshot::channel();
590                    runtime_tx
591                        .send(TheaterCommand::GetActors {
592                            response_tx: cmd_tx,
593                        })
594                        .await?;
595
596                    match cmd_rx.await? {
597                        Ok(actors) => {
598                            info!("Found {} actors", actors.len());
599                            ManagementResponse::ActorList { actors }
600                        }
601                        Err(e) => ManagementResponse::Error {
602                            error: ManagementError::RuntimeError(format!(
603                                "Failed to list actors: {}",
604                                e
605                            )),
606                        },
607                    }
608                }
609                ManagementCommand::SubscribeToActor { id } => {
610                    info!("New subscription request for actor: {:?}", id);
611                    let subscription_id = Uuid::new_v4();
612                    let subscription = Subscription {
613                        id: subscription_id,
614                        client_tx: cmd_client_tx.clone(),
615                    };
616
617                    debug!("Subscription created with ID: {}", subscription_id);
618
619                    // Register the subscription in the global map
620                    subscriptions
621                        .lock()
622                        .await
623                        .entry(id.clone())
624                        .or_default()
625                        .insert(subscription);
626
627                    // Set up the event channel for the subscription
628                    let (event_tx, mut event_rx) = mpsc::channel(32);
629                    runtime_tx
630                        .send(TheaterCommand::SubscribeToActor {
631                            actor_id: id.clone(),
632                            event_tx,
633                        })
634                        .await
635                        .map_err(|e| anyhow::anyhow!("Failed to subscribe: {}", e))?;
636
637                    // Add to the list of subscriptions for this connection
638                    connection_subscriptions.push((id.clone(), subscription_id));
639
640                    // Create a task to forward events to this client
641                    let client_tx_clone = cmd_client_tx.clone();
642                    tokio::spawn(async move {
643                        debug!(
644                            "Starting event forwarder for subscription {}",
645                            subscription_id
646                        );
647                        while let Some(event) = event_rx.recv().await {
648                            debug!("Received event for subscription {}", subscription_id);
649                            let response = match event {
650                                Ok(event) => ManagementResponse::ActorEvent { event },
651                                Err(e) => ManagementResponse::ActorError { error: e },
652                            };
653                            if let Err(e) = client_tx_clone.send(response).await {
654                                debug!("Failed to forward event to client: {}", e);
655                                break;
656                            }
657                        }
658                        debug!(
659                            "Event forwarder for subscription {} stopped",
660                            subscription_id
661                        );
662                    });
663
664                    ManagementResponse::Subscribed {
665                        id,
666                        subscription_id,
667                    }
668                }
669                ManagementCommand::UnsubscribeFromActor {
670                    id,
671                    subscription_id,
672                } => {
673                    debug!(
674                        "Removing subscription {} for actor {:?}",
675                        subscription_id, id
676                    );
677
678                    // Remove subscription from the tracking list for this connection
679                    connection_subscriptions
680                        .retain(|(aid, sid)| *aid != id || *sid != subscription_id);
681
682                    // Remove from the global subscriptions map
683                    let mut subs = subscriptions.lock().await;
684                    if let Some(actor_subs) = subs.get_mut(&id) {
685                        actor_subs.retain(|sub| sub.id != subscription_id);
686
687                        // Remove the entry if no subscriptions remain
688                        if actor_subs.is_empty() {
689                            subs.remove(&id);
690                        }
691                    }
692
693                    debug!("Subscription removed");
694                    ManagementResponse::Unsubscribed { id }
695                }
696                ManagementCommand::SendActorMessage { id, data } => {
697                    info!("Sending message to actor: {:?}", id);
698                    runtime_tx
699                        .send(TheaterCommand::SendMessage {
700                            actor_id: id.clone(),
701                            actor_message: ActorMessage::Send(ActorSend { data: data.clone() }),
702                        })
703                        .await?;
704
705                    ManagementResponse::SentMessage { id }
706                }
707                ManagementCommand::RequestActorMessage { id, data } => {
708                    info!("Requesting message from actor: {:?}", id);
709                    let (cmd_tx, cmd_rx) = tokio::sync::oneshot::channel();
710                    runtime_tx
711                        .send(TheaterCommand::SendMessage {
712                            actor_id: id.clone(),
713                            actor_message: ActorMessage::Request(ActorRequest {
714                                data: data.clone(),
715                                response_tx: cmd_tx,
716                            }),
717                        })
718                        .await?;
719
720                    let response = cmd_rx.await?;
721                    ManagementResponse::RequestedMessage {
722                        id,
723                        message: response,
724                    }
725                }
726                ManagementCommand::GetActorManifest { id } => {
727                    info!("Getting manifest for actor: {:?}", id);
728                    let (cmd_tx, cmd_rx) = tokio::sync::oneshot::channel();
729                    runtime_tx
730                        .send(TheaterCommand::GetActorManifest {
731                            actor_id: id.clone(),
732                            response_tx: cmd_tx,
733                        })
734                        .await?;
735
736                    let manifest = cmd_rx.await?;
737                    ManagementResponse::ActorManifest {
738                        id,
739                        manifest: manifest?,
740                    }
741                }
742                ManagementCommand::GetActorStatus { id } => {
743                    info!("Getting status for actor: {:?}", id);
744                    let (cmd_tx, cmd_rx) = tokio::sync::oneshot::channel();
745                    runtime_tx
746                        .send(TheaterCommand::GetActorStatus {
747                            actor_id: id.clone(),
748                            response_tx: cmd_tx,
749                        })
750                        .await?;
751
752                    let status = cmd_rx.await?;
753                    ManagementResponse::ActorStatus {
754                        id,
755                        status: status?,
756                    }
757                }
758                ManagementCommand::RestartActor { id } => {
759                    info!("Restarting actor: {:?}", id);
760                    let (cmd_tx, cmd_rx) = tokio::sync::oneshot::channel();
761                    runtime_tx
762                        .send(TheaterCommand::RestartActor {
763                            actor_id: id.clone(),
764                            response_tx: cmd_tx,
765                        })
766                        .await?;
767
768                    match cmd_rx.await? {
769                        Ok(_) => ManagementResponse::Restarted { id },
770                        Err(e) => ManagementResponse::Error {
771                            error: ManagementError::RuntimeError(format!(
772                                "Failed to restart actor: {}",
773                                e
774                            )),
775                        },
776                    }
777                }
778                ManagementCommand::GetActorState { id } => {
779                    info!("Getting state for actor: {:?}", id);
780                    let (cmd_tx, cmd_rx) = tokio::sync::oneshot::channel();
781                    runtime_tx
782                        .send(TheaterCommand::GetActorState {
783                            actor_id: id.clone(),
784                            response_tx: cmd_tx,
785                        })
786                        .await?;
787
788                    let state = cmd_rx.await?;
789                    ManagementResponse::ActorState { id, state: state? }
790                }
791                ManagementCommand::GetActorEvents { id } => {
792                    info!("Getting events for actor: {:?}", id);
793                    let (cmd_tx, cmd_rx) = tokio::sync::oneshot::channel();
794                    runtime_tx
795                        .send(TheaterCommand::GetActorEvents {
796                            actor_id: id.clone(),
797                            response_tx: cmd_tx,
798                        })
799                        .await?;
800
801                    match cmd_rx.await {
802                        Ok(result) => match result {
803                            Ok(events) => {
804                                debug!(
805                                    "Successfully retrieved {} events for actor {}",
806                                    events.len(),
807                                    id
808                                );
809                                ManagementResponse::ActorEvents { id, events }
810                            }
811                            Err(e) => {
812                                debug!("Error getting events for actor {}: {}", id, e);
813                                ManagementResponse::Error { error: e.into() }
814                            }
815                        },
816                        Err(e) => {
817                            error!("Failed to receive events response: {}", e);
818                            ManagementResponse::Error {
819                                error: ManagementError::CommunicationError(format!(
820                                    "Failed to receive events response: {}",
821                                    e
822                                )),
823                            }
824                        }
825                    }
826                }
827                ManagementCommand::GetActorMetrics { id } => {
828                    info!("Getting metrics for actor: {:?}", id);
829                    let (cmd_tx, cmd_rx) = tokio::sync::oneshot::channel();
830                    runtime_tx
831                        .send(TheaterCommand::GetActorMetrics {
832                            actor_id: id.clone(),
833                            response_tx: cmd_tx,
834                        })
835                        .await?;
836
837                    let metrics = cmd_rx.await?;
838                    ManagementResponse::ActorMetrics {
839                        id,
840                        metrics: serde_json::to_value(metrics?)?,
841                    }
842                }
843                ManagementCommand::UpdateActorComponent { id, component } => {
844                    info!("Updating component for actor {:?} to {}", id, component);
845                    let (cmd_tx, cmd_rx) = tokio::sync::oneshot::channel();
846                    runtime_tx
847                        .send(TheaterCommand::UpdateActorComponent {
848                            actor_id: id.clone(),
849                            component: component.clone(),
850                            response_tx: cmd_tx,
851                        })
852                        .await?;
853
854                    match cmd_rx.await? {
855                        Ok(_) => ManagementResponse::ActorComponentUpdated { id },
856                        Err(e) => ManagementResponse::Error {
857                            error: ManagementError::RuntimeError(format!(
858                                "Failed to update actor component: {}",
859                                e
860                            )),
861                        },
862                    }
863                }
864                // Handle channel management commands
865                ManagementCommand::OpenChannel {
866                    actor_id,
867                    initial_message,
868                } => {
869                    info!("Opening channel to actor: {:?}", actor_id);
870
871                    // Create a response channel
872                    let (response_tx, response_rx) = tokio::sync::oneshot::channel();
873
874                    // Generate a channel ID
875                    let client_id = ChannelParticipant::External;
876                    let channel_id = ChannelId::new(&client_id, &actor_id);
877                    let channel_id_str = channel_id.0.clone();
878
879                    // Send the channel open command to the runtime
880                    runtime_tx
881                        .send(TheaterCommand::ChannelOpen {
882                            initiator_id: client_id.clone(),
883                            target_id: actor_id.clone(),
884                            channel_id: channel_id.clone(),
885                            initial_message,
886                            response_tx,
887                        })
888                        .await
889                        .map_err(|e| {
890                            anyhow::anyhow!("Failed to send channel open command: {}", e)
891                        })?;
892
893                    // Wait for the response
894                    match response_rx.await {
895                        Ok(result) => {
896                            match result {
897                                Ok(accepted) => {
898                                    if accepted {
899                                        // Channel opened successfully
900                                        info!("Channel opened successfully: {}", channel_id_str);
901
902                                        // Register the channel subscription to receive messages
903                                        let channel_sub = ChannelSubscription {
904                                            channel_id: channel_id_str.clone(),
905                                            initiator_id: client_id.clone(),
906                                            target_id: actor_id.clone(),
907                                            client_tx: cmd_client_tx.clone(),
908                                        };
909
910                                        channel_subscriptions
911                                            .lock()
912                                            .await
913                                            .insert(channel_id_str.clone(), channel_sub);
914
915                                        // Track this channel for cleanup on disconnect
916                                        connection_channel_subscriptions
917                                            .push(channel_id_str.clone());
918
919                                        ManagementResponse::ChannelOpened {
920                                            channel_id: channel_id_str,
921                                            actor_id,
922                                        }
923                                    } else {
924                                        // Channel rejected by target
925                                        ManagementResponse::Error {
926                                            error: ManagementError::ChannelRejected,
927                                        }
928                                    }
929                                }
930                                Err(e) => ManagementResponse::Error {
931                                    error: ManagementError::RuntimeError(format!(
932                                        "Error opening channel: {}",
933                                        e
934                                    )),
935                                },
936                            }
937                        }
938                        Err(e) => ManagementResponse::Error {
939                            error: ManagementError::CommunicationError(format!(
940                                "Failed to receive channel open response: {}",
941                                e
942                            )),
943                        },
944                    }
945                }
946                ManagementCommand::SendOnChannel {
947                    channel_id,
948                    message,
949                } => {
950                    info!("Sending message on channel: {}", channel_id);
951
952                    // Parse the channel ID
953                    let channel_id_parsed = ChannelId(channel_id.clone());
954                    let client_id = ChannelParticipant::External;
955
956                    // Send the message on the channel
957                    runtime_tx
958                        .send(TheaterCommand::ChannelMessage {
959                            channel_id: channel_id_parsed,
960                            message,
961                            sender_id: client_id,
962                        })
963                        .await
964                        .map_err(|e| anyhow::anyhow!("Failed to send message on channel: {}", e))?;
965
966                    ManagementResponse::MessageSent { channel_id }
967                }
968                ManagementCommand::CloseChannel { channel_id } => {
969                    info!("Closing channel: {}", channel_id);
970
971                    // Parse the channel ID
972                    let channel_id_parsed = ChannelId(channel_id.clone());
973
974                    // Close the channel
975                    runtime_tx
976                        .send(TheaterCommand::ChannelClose {
977                            channel_id: channel_id_parsed,
978                        })
979                        .await
980                        .map_err(|e| anyhow::anyhow!("Failed to close channel: {}", e))?;
981
982                    // Remove from channel subscriptions
983                    channel_subscriptions.lock().await.remove(&channel_id);
984                    connection_channel_subscriptions.retain(|id| id != &channel_id);
985
986                    ManagementResponse::ChannelClosed { channel_id }
987                }
988                ManagementCommand::NewStore {} => {
989                    info!("Creating new store");
990                    let (cmd_tx, cmd_rx) = tokio::sync::oneshot::channel();
991                    runtime_tx
992                        .send(TheaterCommand::NewStore {
993                            response_tx: cmd_tx,
994                        })
995                        .await?;
996
997                    let store_id = cmd_rx.await?;
998                    ManagementResponse::StoreCreated {
999                        store_id: store_id?.id,
1000                    }
1001                }
1002            };
1003
1004            debug!("Sending response: {:?}", response);
1005            if let Err(e) = client_tx.send(response).await {
1006                error!("Failed to send response: {}", e);
1007                break;
1008            }
1009            debug!("Response sent");
1010        }
1011
1012        // Clean up all subscriptions for this connection
1013        debug!(
1014            "Connection closed, cleaning up {} subscriptions",
1015            connection_subscriptions.len()
1016        );
1017        let mut subs = subscriptions.lock().await;
1018
1019        for (actor_id, sub_id) in connection_subscriptions {
1020            if let Some(actor_subs) = subs.get_mut(&actor_id) {
1021                actor_subs.retain(|sub| sub.id != sub_id);
1022
1023                // Remove the entry if no subscriptions remain
1024                if actor_subs.is_empty() {
1025                    subs.remove(&actor_id);
1026                }
1027            }
1028        }
1029
1030        // Clean up channel subscriptions
1031        debug!(
1032            "Connection closed, cleaning up {} channel subscriptions",
1033            connection_channel_subscriptions.len()
1034        );
1035        let mut channel_subs = channel_subscriptions.lock().await;
1036
1037        for channel_id in connection_channel_subscriptions {
1038            channel_subs.remove(&channel_id);
1039        }
1040
1041        debug!("Cleaned up all subscriptions for the connection");
1042        Ok(())
1043    }
1044}
