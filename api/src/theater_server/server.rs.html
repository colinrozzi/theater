1use anyhow::Result;
2use std::path::PathBuf;
3use theater::logging;
4use theater_server::TheaterServer;
5use tracing::info;
6
7use crate::args::ServerArgs;
8
9pub async fn start_server(args: &ServerArgs) -> Result<()> {
10    // Create the runtime log file path
11    let log_path = shellexpand::env(&args.log_dir)
12        .map_err(|e| anyhow::anyhow!("Failed to expand log directory: {}", e))?;
13    let log_path = PathBuf::from(log_path.as_ref()).join("theater_server.log");
14
15    let log_level = args.log_level.parse().unwrap_or_else(|_| {
16        eprintln!(
17            "Invalid log level: {}. Defaulting to 'info'.",
18            args.log_level
19        );
20        tracing::Level::INFO
21    });
22
23    logging::setup_global_logging(log_path, &log_level, args.log_stdout)
24        .expect("Failed to setup logging");
25
26    info!("Starting theater server on {}", args.address);
27    info!("Logging to directory: {}", args.log_dir);
28
29    // Create and run the theater server
30    let server = TheaterServer::new(args.address).await?;
31    server.run().await?;
32
33    Ok(())
34}
