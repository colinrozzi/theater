2//! # Actor Runtime
3//!
4//! The Actor Runtime is responsible for initializing, running, and managing the lifecycle
5//! of WebAssembly actors within the Theater system. It coordinates the various components
6//! that an actor needs to function, including handlers, the executor, and communication channels.
7//!
8//! ## Purpose
9//!
10//! The Actor Runtime serves as the glue between the Theater system and individual actors by:
11//!
12//! - Managing the full lifecycle of an actor from instantiation to shutdown
13//! - Setting up host function handlers that provide capabilities to the actor
14//! - Initializing the actor with its configuration and state
15//! - Propagating shutdown signals from parent to child components
16//! - Coordinating the execution of all components that make up an actor's environment
17//!
18//! Each actor in the system has its own dedicated runtime, providing isolation and independent
19//! management of resources.

51/// # ActorRuntime
52///
53/// Coordinates the execution and lifecycle of a single WebAssembly actor within the Theater system.
54///
55/// ## Purpose
56///
57/// `ActorRuntime` manages the various components that make up an actor's execution environment,
58/// including the executor, handlers, and communication channels. It's responsible for starting
59/// the actor, setting up its capabilities via handlers, and ensuring proper shutdown.
60///
61/// ## Example
62///
63/// ```rust
64/// use theater::actor_runtime::ActorRuntime;
65/// use theater::config::ManifestConfig;
66/// use theater::id::TheaterId;
67/// use tokio::sync::mpsc;
68///
69/// async fn start_actor(
70///     id: TheaterId,
71///     config: &ManifestConfig,
72///     theater_tx: mpsc::Sender<TheaterCommand>,
73///     shutdown_receiver: ShutdownReceiver,
74/// ) -> Result<(), Box<dyn std::error::Error>> {
75///     // Create the necessary channels
76///     let (actor_sender, _) = mpsc::channel(32);
77///     let (_, actor_mailbox) = mpsc::channel(32);
78///     let (operation_tx, operation_rx) = mpsc::channel(32);
79///     let (response_tx, mut response_rx) = mpsc::channel(1);
80///     
81///     // Start the actor runtime
82///     let mut runtime = ActorRuntime::start(
83///         id.clone(),
84///         config,
85///         None, // No initial state
86///         theater_tx,
87///         actor_sender,
88///         actor_mailbox,
89///         operation_rx,
90///         operation_tx,
91///         true, // Initialize the actor
92///         shutdown_receiver,
93///         response_tx,
94///     ).await?;
95///     
96///     // Wait for the start result
97///     match response_rx.recv().await {
98///         Some(StartActorResult::Success(actor_id)) => {
99///             println!("Actor started successfully: {}", actor_id);
100///         }
101///         Some(StartActorResult::Failure(actor_id, error)) => {
102///             println!("Actor {} failed to start: {}", actor_id, error);
103///             return Err(error.into());
104///         }
105///         None => {
106///             println!("No response received from actor start");
107///             return Err("No response".into());
108///         }
109///     }
110///     
111///     // Later, stop the actor
112///     runtime.stop().await?;
113///     
114///     Ok(())
115/// }
116/// ```
117///
118/// ## Safety
119///
120/// While `ActorRuntime` itself is safe to use, it manages WebAssembly execution which is
121/// inherently unsafe. The runtime ensures isolation between actors and proper handling of 
122/// errors from WebAssembly code.
123///
124/// ## Security
125///
126/// The runtime enforces security boundaries by:
127/// - Setting up appropriate handler capabilities based on the actor's configuration
128/// - Ensuring proper isolation between actors
129/// - Managing resource allocation and cleanup
130/// - Recording events for auditing
131///
132/// ## Implementation Notes
133///
134/// The runtime uses Tokio tasks for concurrent execution of the actor's components:
135/// - One task for the actor executor
136/// - Separate tasks for each handler
137/// - A monitoring task for shutdown propagation
138pub struct ActorRuntime {
139    /// Unique identifier for this actor
140    pub actor_id: TheaterId,
141    /// Handles to the running handler tasks
142    handler_tasks: Vec<JoinHandle<()>>,
143    /// Handle to the actor executor task
144    actor_executor_task: JoinHandle<()>,
145    /// Controller for graceful shutdown of all components
146    shutdown_controller: ShutdownController,
147}

149/// # Result of starting an actor
150///
151/// Represents the outcome of attempting to start an actor.
152///
153/// ## Purpose
154///
155/// This enum provides detailed information about whether an actor was successfully
156/// started or encountered errors during initialization. It includes the actor's ID
157/// in both success and failure cases, and detailed error information in the failure case.
158///
159/// ## Example
160///
161/// ```rust
162/// use theater::actor_runtime::StartActorResult;
163/// use theater::id::TheaterId;
164///
165/// fn handle_start_result(result: StartActorResult) {
166///     match result {
167///         StartActorResult::Success(id) => {
168///             println!("Actor {} started successfully", id);
169///             // Proceed with using the actor
170///         }
171///         StartActorResult::Failure(id, error) => {
172///             eprintln!("Actor {} failed to start: {}", id, error);
173///             // Handle the error or try recovery
174///         }
175///     }
176/// }
177/// ```
178#[derive(Debug)]
179pub enum StartActorResult {
180    /// Actor successfully started
181    Success(TheaterId),
182    /// Actor failed to start with error message
183    Failure(TheaterId, String),
184}

186impl ActorRuntime {
187    /// # Start a new actor runtime
188    ///
189    /// Initializes and starts an actor runtime with the specified configuration,
190    /// setting up all necessary components for the actor to run.
191    ///
192    /// ## Parameters
193    ///
194    /// * `id` - Unique identifier for the actor
195    /// * `config` - Configuration for the actor from its manifest
196    /// * `state_bytes` - Optional initial state for the actor
197    /// * `theater_tx` - Channel for sending commands back to the Theater runtime
198    /// * `actor_sender` - Channel for sending messages to the actor
199    /// * `actor_mailbox` - Channel for receiving messages from other actors
200    /// * `operation_rx` - Channel for receiving operations to perform
201    /// * `operation_tx` - Channel for sending operations to the executor
202    /// * `init` - Whether to initialize the actor (call the init function)
203    /// * `parent_shutdown_receiver` - Receiver for shutdown signals from the parent
204    /// * `response_tx` - Channel for sending the start result back to the caller
205    ///
206    /// ## Returns
207    ///
208    /// * `Ok(ActorRuntime)` - The started actor runtime
209    /// * `Err(anyhow::Error)` - Error that occurred during startup
210    ///
211    /// ## Implementation Notes
212    ///
213    /// This method performs the following steps:
214    /// 1. Creates the actor store for state management
215    /// 2. Records the actor load event in the chain
216    /// 3. Sets up handlers based on the configuration
217    /// 4. Creates and instantiates the WebAssembly component
218    /// 5. Initializes the actor if requested
219    /// 6. Starts the executor and handlers
220    /// 7. Sets up shutdown signal propagation
221    ///
222    /// The method is asynchronous and returns once the actor is fully initialized
223    /// and ready to accept operations.
224    pub async fn start(
225        id: TheaterId,
226        config: &ManifestConfig,
227        state_bytes: Option<Vec<u8>>,
228        theater_tx: Sender<TheaterCommand>,
229        actor_sender: Sender<ActorMessage>,
230        actor_mailbox: Receiver<ActorMessage>,
231        operation_rx: Receiver<ActorOperation>,
232        operation_tx: Sender<ActorOperation>,
233        init: bool,
234        parent_shutdown_receiver: ShutdownReceiver,
235        response_tx: Sender<StartActorResult>,
236    ) -> Result<Self> {
237        let actor_handle = ActorHandle::new(operation_tx.clone());
238        let actor_store = ActorStore::new(id.clone(), theater_tx.clone(), actor_handle.clone());
239
240        let manifest_store = ContentStore::from_id("manifest");
241        let manifest_id = manifest_store
242            .store(config.clone().into_fixed_bytes())
243            .await?;
244
245        actor_store.record_event(ChainEventData {
246            event_type: "theater-runtime".to_string(),
247            data: EventData::TheaterRuntime(TheaterRuntimeEventData::ActorLoadCall {
248                manifest_id: manifest_id.clone(),
249            }),
250            timestamp: chrono::Utc::now().timestamp_millis() as u64,
251            description: format!("Loading actor [{}] from manifest [{}] ", id, manifest_id).into(),
252        });
253
254        // Create a local shutdown controller for this runtime
255        let (shutdown_controller, _) = ShutdownController::new();
256        let mut handlers = Vec::new();
257
258        handlers.push(Handler::MessageServer(MessageServerHost::new(
259            actor_sender,
260            actor_mailbox,
261            theater_tx.clone(),
262        )));
263
264        for handler_config in &config.handlers {
265            let handler = match handler_config {
266                HandlerConfig::MessageServer(_) => {
267                    panic!("MessageServer handler is already added")
268                }
269                HandlerConfig::FileSystem(config) => {
270                    Handler::FileSystem(FileSystemHost::new(config.clone()))
271                }
272                HandlerConfig::HttpClient(config) => {
273                    Handler::HttpClient(HttpClientHost::new(config.clone()))
274                }
275                HandlerConfig::HttpFramework(_) => Handler::HttpFramework(HttpFramework::new()),
276                HandlerConfig::Runtime(config) => {
277                    Handler::Runtime(RuntimeHost::new(config.clone()))
278                }
279                HandlerConfig::Supervisor(config) => {
280                    Handler::Supervisor(SupervisorHost::new(config.clone()))
281                }
282                HandlerConfig::Store(config) => Handler::Store(StoreHost::new(config.clone())),
283                HandlerConfig::Timing(config) => Handler::Timing(TimingHost::new(config.clone())),
284            };
285            handlers.push(handler);
286        }
287
288        let mut actor_component = match ActorComponent::new(config, actor_store).await {
289            Ok(component) => component,
290            Err(e) => {
291                let error_message = format!(
292                    "Failed to create actor component for actor {}: {}",
293                    config.name, e
294                );
295                error!("{}", error_message);
296                // Send failure result back to spawner with detailed error message
297                if let Err(send_err) = response_tx
298                    .send(StartActorResult::Failure(id.clone(), error_message))
299                    .await
300                {
301                    error!("Failed to send failure response: {}", send_err);
302                }
303                return Err(e.into());
304            }
305        };
306
307        // Add the host functions to the linker of the actor
308        {
309            for handler in &mut handlers {
310                info!(
311                    "Setting up host functions for handler: {:?}",
312                    handler.name()
313                );
314                match handler.setup_host_functions(&mut actor_component).await {
315                    Ok(_) => {} // Successfully set up host functions
316                    Err(e) => {
317                        let error_message = format!(
318                            "Failed to set up host functions for handler {}: {}",
319                            handler.name(),
320                            e
321                        );
322                        error!("{}", error_message);
323                        // Send failure result back to spawner with detailed error message
324                        if let Err(send_err) = response_tx
325                            .send(StartActorResult::Failure(id.clone(), error_message))
326                            .await
327                        {
328                            error!("Failed to send failure response: {}", send_err);
329                        }
330                        return Err(e);
331                    }
332                }
333            }
334        }
335
336        let mut actor_instance = match actor_component.instantiate().await {
337            Ok(instance) => instance,
338            Err(e) => {
339                let error_message = format!("Failed to instantiate actor {}: {}", id, e);
340                error!("{}", error_message);
341                // Send failure result back to spawner with detailed error message
342                if let Err(send_err) = response_tx
343                    .send(StartActorResult::Failure(id.clone(), error_message))
344                    .await
345                {
346                    error!("Failed to send failure response: {}", send_err);
347                }
348                return Err(e.into());
349            }
350        };
351
352        {
353            for handler in &handlers {
354                info!("Creating functions for handler: {:?}", handler.name());
355                match handler.add_export_functions(&mut actor_instance).await {
356                    Ok(_) => {} // Successfully added export functions
357                    Err(e) => {
358                        let error_message = format!(
359                            "Failed to create export functions for handler {}: {}",
360                            handler.name(),
361                            e
362                        );
363                        error!("{}", error_message);
364                        // Send failure result back to spawner with detailed error message
365                        if let Err(send_err) = response_tx
366                            .send(StartActorResult::Failure(id.clone(), error_message))
367                            .await
368                        {
369                            error!("Failed to send failure response: {}", send_err);
370                        }
371                        return Err(e.into());
372                    }
373                }
374            }
375        }
376
377        // Actor handle already created above
378        let mut init_state = None;
379        if init {
380            info!("Loading init state for actor: {:?}", id);
381
382            // Get state from config if available
383            let config_state = config.load_init_state().unwrap_or(None);
384
385            // Merge with provided state
386            init_state = crate::utils::merge_initial_states(config_state, state_bytes)
387                .expect("Failed to merge initial states");
388
389            info!("Final init state ready: {:?}", init_state.is_some());
390        }
391
392        actor_instance.store.data_mut().set_state(init_state);
393
394        // Create a shutdown receiver for the executor
395        let executor_shutdown = shutdown_controller.subscribe();
396        let mut actor_executor =
397            ActorExecutor::new(actor_instance, operation_rx, executor_shutdown, theater_tx);
398        let executor_task = tokio::spawn(async move { actor_executor.run().await });
399
400        // Notify the caller that the actor has started
401        if let Err(e) = response_tx
402            .send(StartActorResult::Success(id.clone()))
403            .await
404        {
405            error!("Failed to send success response: {}", e);
406            // Even though we couldn't send the response, we'll return the runtime
407            // since it's been initialized successfully
408        }
409
410        if init {
411            match actor_handle
412                .call_function::<(String,), ()>(
413                    "ntwk:theater/actor.init".to_string(),
414                    (id.to_string(),),
415                )
416                .await
417            {
418                Ok(_) => {} // Successfully called init function
419                Err(e) => {
420                    let error_message =
421                        format!("Failed to call init function for actor {}: {}", id, e);
422                    error!("{}", error_message);
423                    // Send failure result back to spawner with detailed error message
424                    if let Err(send_err) = response_tx
425                        .send(StartActorResult::Failure(id.clone(), error_message.clone()))
426                        .await
427                    {
428                        error!("Failed to send failure response: {}", send_err);
429                    }
430                    return Err(anyhow::anyhow!(error_message));
431                }
432            }
433        }
434
435        let mut handler_tasks: Vec<JoinHandle<()>> = Vec::new();
436
437        for mut handler in handlers {
438            info!("Starting handler: {:?}", handler.name());
439            let actor_handle = actor_handle.clone();
440            let handler_shutdown = shutdown_controller.subscribe();
441            let handler_task = tokio::spawn(async move {
442                if let Err(e) = handler.start(actor_handle, handler_shutdown).await {
443                    warn!("Handler failed: {:?}", e);
444                }
445            });
446            handler_tasks.push(handler_task);
447        }
448
449        // Monitor parent shutdown signal and propagate
450        let shutdown_controller_clone = shutdown_controller.clone();
451        let mut parent_shutdown_receiver_clone = parent_shutdown_receiver;
452        tokio::spawn(async move {
453            debug!("Actor waiting for parent shutdown signal");
454            parent_shutdown_receiver_clone.wait_for_shutdown().await;
455            info!("Actor runtime received parent shutdown signal");
456            debug!("Propagating shutdown signal to all handler components");
457            shutdown_controller_clone.signal_shutdown();
458            debug!("Shutdown signal propagated to all components");
459        });
460
461        Ok(ActorRuntime {
462            actor_id: id.clone(),
463            handler_tasks,
464            actor_executor_task: executor_task,
465            shutdown_controller,
466        })
467    }
468
469    /// # Stop the actor runtime
470    ///
471    /// Gracefully shuts down the actor runtime and all its components,
472    /// including the executor and handlers.
473    ///
474    /// ## Returns
475    ///
476    /// * `Ok(())` - The runtime was successfully shut down
477    /// * `Err(anyhow::Error)` - An error occurred during shutdown
478    ///
479    /// ## Example
480    ///
481    /// ```rust
482    /// async fn stop_actor(runtime: &mut ActorRuntime) -> Result<(), Box<dyn std::error::Error>> {
483    ///     // Signal shutdown to all components and wait for completion
484    ///     runtime.stop().await?;
485    ///     println!("Actor has been shut down");
486    ///     Ok(())
487    /// }
488    /// ```
489    ///
490    /// ## Implementation Notes
491    ///
492    /// This method performs the following steps:
493    /// 1. Signals shutdown to all components via the shutdown controller
494    /// 2. Waits briefly for tasks to shut down gracefully
495    /// 3. Forcefully aborts any tasks that didn't shut down gracefully
496    ///
497    /// The method is designed to ensure that resources are properly cleaned up
498    /// even if some components fail to shut down gracefully.
499    pub async fn stop(&mut self) -> Result<()> {
500        info!("Initiating actor runtime shutdown");
501
502        // Signal shutdown to all components
503        info!("Signaling shutdown to all components");
504        self.shutdown_controller.signal_shutdown();
505
506        // Wait a bit for graceful shutdown
507        tokio::time::sleep(std::time::Duration::from_millis(200)).await;
508
509        // If any handlers are still running, abort them
510        for task in self.handler_tasks.drain(..) {
511            if !task.is_finished() {
512                debug!("Aborting handler task that didn't shut down gracefully");
513                task.abort();
514            }
515        }
516
517        // Finally abort the executor if it's still running
518        if !self.actor_executor_task.is_finished() {
519            debug!("Aborting executor task that didn't shut down gracefully");
520            self.actor_executor_task.abort();
521        }
522
523        info!("Actor runtime shutdown complete");
524        Ok(())
525    }
