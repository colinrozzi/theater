19/// # ActorError
20///
21/// Represents errors that can occur during actor execution.
22///
23/// This enum provides detailed error information for various failure modes that
24/// might occur when interacting with an actor. These errors are propagated back
25/// to callers to help diagnose and handle problems.
26#[derive(
27    Error, Debug, Clone, ComponentType, Lift, Lower, Serialize, Deserialize, PartialEq, Hash, Eq,
28)]
29#[component(variant)]
30pub enum ActorError {
31    /// Operation exceeded the maximum allowed execution time
32    #[error("Operation timed out after {0:?}")]
33    #[component(name = "operation-timeout")]
34    OperationTimeout(u64),
35
36    /// Communication channel to the actor was closed unexpectedly
37    #[error("Operation channel closed")]
38    #[component(name = "channel-closed")]
39    ChannelClosed,
40
41    /// Actor is in the process of shutting down and cannot accept new operations
42    #[error("Actor is shutting down")]
43    #[component(name = "shutting-down")]
44    ShuttingDown,
45
46    /// The requested WebAssembly function was not found in the actor
47    #[error("Function not found: {0}")]
48    #[component(name = "function-not-found")]
49    FunctionNotFound(String),
50
51    /// Parameter or return types did not match the WebAssembly function signature
52    #[error("Type mismatch for function {0}")]
53    #[component(name = "type-mismatch")]
54    TypeMismatch(String),
55
56    /// An internal error occurred during execution
57    #[error("Internal error: {0}")]
58    #[component(name = "internal-error")]
59    Internal(ChainEvent),
60
61    /// Unexpected error
62    #[error("Unexpected error: {0}")]
63    #[component(name = "unexpected-error")]
64    UnexpectedError(String),
65
66    /// Failed to serialize or deserialize data
67    #[error("Serialization error")]
68    #[component(name = "serialization-error")]
69    SerializationError,
70
71    /// Failed to update the actor's WebAssembly component
72    #[error("Failed to update component: {0}")]
73    #[component(name = "update-component-error")]
74    UpdateComponentError(String),
75
76    /// Actor is paused
77    #[error("Actor is paused")]
78    #[component(name = "actor-paused")]
79    Paused,
80
81    /// Actor is not paused
82    #[error("Actor is not paused")]
83    #[component(name = "actor-not-paused")]
84    NotPaused,
85    // Failed to update actor due to permission or validation error
86    //#[error("Actor update failed: {0}")]
87    //#[component(name = "update-error")]
88    //UpdateError(String),
89    #[error("Handler error: {0}")]
90    #[component(name = "handler-error")]
91    HandlerError(String),
92}

102#[derive(Debug, Clone, ComponentType, Lift, Lower, Serialize, Deserialize, Copy)]
103#[component(enum)]
104#[repr(u8)]
105pub enum WitErrorType {
106    #[component(name = "operation-timeout")]
107    OperationTimeout,
108    #[component(name = "channel-closed")]
109    ChannelClosed,
110    #[component(name = "shutting-down")]
111    ShuttingDown,
112    #[component(name = "function-not-found")]
113    FunctionNotFound,
114    #[component(name = "type-mismatch")]
115    TypeMismatch,
116    #[component(name = "internal")]
117    Internal,
118    #[component(name = "serialization-error")]
119    SerializationError,
120    #[component(name = "update-component-error")]
121    UpdateComponentError,
122    #[component(name = "paused")]
123    Paused,
124    //#[component(name = "update-error")]
125    //UpdateError,
126}

128impl From<ActorError> for WitActorError {
129    fn from(error: ActorError) -> Self {
130        let (error_type, data) = match error {
131            ActorError::OperationTimeout(data) => (
132                WitErrorType::OperationTimeout,
133                Some(data.to_le_bytes().to_vec()),
134            ),
135            ActorError::ChannelClosed => (WitErrorType::ChannelClosed, None),
136            ActorError::ShuttingDown => (WitErrorType::ShuttingDown, None),
137            ActorError::FunctionNotFound(data) => {
138                (WitErrorType::FunctionNotFound, Some(data.into_bytes()))
139            }
140            ActorError::TypeMismatch(data) => (WitErrorType::TypeMismatch, Some(data.into_bytes())),
141            ActorError::Internal(data) => (
142                WitErrorType::Internal,
143                Some(serde_json::to_vec(&data).unwrap()),
144            ),
145            ActorError::SerializationError => (WitErrorType::SerializationError, None),
146            ActorError::UpdateComponentError(data) => {
147                (WitErrorType::UpdateComponentError, Some(data.into_bytes()))
148            }
149            ActorError::Paused => (WitErrorType::Paused, None),
150            ActorError::NotPaused => (WitErrorType::Paused, None),
151            ActorError::UnexpectedError(data) => (WitErrorType::Internal, Some(data.into_bytes())),
152            ActorError::HandlerError(data) => (WitErrorType::Internal, Some(data.into_bytes())),
153            //        ActorError::UpdateError(data) => (WitErrorType::UpdateError, Some(data.into_bytes())),
154        };
155        Self { error_type, data }
156    }
157}

159/// # ActorOperation
160///
161/// Represents the different types of operations that can be performed on an actor.
162///
163/// This enum defines the message types that can be sent to an `ActorRuntime` via
164/// its operation channel. Each variant includes the necessary data for the operation
165/// and a oneshot channel sender for returning the result.
166#[derive(Debug)]
167pub enum ActorOperation {
168    /// Call a WebAssembly function in the actor
169    CallFunction {
170        /// Name of the function to call
171        name: String,
172        /// Serialized parameters for the function
173        params: Vec<u8>,
174        /// Channel to send the result back to the caller
175        response_tx: oneshot::Sender<Result<Vec<u8>, ActorError>>,
176    },
177    /// Update a WebAssembly component in the actor
178    UpdateComponent {
179        /// Address of the component to update
180        component_address: String,
181        /// Channel to send the result back to the caller
182        response_tx: oneshot::Sender<Result<(), ActorError>>,
183    },
184}

186#[derive(Debug)]
187pub enum ActorControl {
188    /// Pause the actor
189    Pause {
190        /// Channel to confirm pause completion

193    /// Resume the actor
194    Resume {
195        /// Channel to confirm resume completion
196        response_tx: oneshot::Sender<Result<(), ActorError>>,
197    },
198    /// Initiate actor shutdown
199    Shutdown {
200        /// Channel to confirm shutdown completion
201        response_tx: oneshot::Sender<Result<(), ActorError>>,
202    },
203    /// Forcefully terminate the actor, aborting any ongoing operations
204    Terminate {
205        /// Channel to confirm termination completion
206        response_tx: oneshot::Sender<Result<(), ActorError>>,
207    },
208}
