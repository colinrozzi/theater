4/// # Chain Event Data
5///
6/// `ChainEventData` is the base structure for all typed events in the Theater system.
7/// It wraps specific event data with common metadata like event type, timestamp, and
8/// optional human-readable description.
9///
10/// ## Purpose
11///
12/// This struct serves as the bridge between strongly-typed event data and the generic
13/// chain event system. It provides a standardized way to attach metadata to any type
14/// of event, ensuring that all events in the system have consistent properties
15/// regardless of their specific payload type.
16///
17/// ## Example
18///
19/// ```rust
20/// use theater::events::{ChainEventData, EventData};
21/// use theater::events::runtime::RuntimeEventData;
22/// use chrono::Utc;
23///
24/// // Create a runtime initialization event
25/// let event_data = ChainEventData {
26///     event_type: "actor.init".to_string(),
27///     data: EventData::Runtime(RuntimeEventData::InitCall {
28///         params: "init params".to_string(),
29///     }),
30///     timestamp: Utc::now().timestamp() as u64,
31///     description: Some("Actor initialized".to_string()),
32/// };
33/// ```
34///
35/// ## Implementation Notes
36///
37/// The `ChainEventData` structure is designed to be serializable, allowing events
38/// to be stored, transmitted over the network, or saved to a file. It can be
39/// converted to a `ChainEvent` for inclusion in an actor's event chain using
40/// the `to_chain_event` method.
41#[derive(Debug, Clone, Serialize, Deserialize)]
42pub struct ChainEventData {
43    /// The type identifier for this event, used for filtering and routing.
44    /// This should be a dot-separated string like "subsystem.action".
45    pub event_type: String,
46    /// The specific event data payload, containing domain-specific information.
47    pub data: EventData,
48    /// Unix timestamp (in seconds) when the event was created.
49    pub timestamp: u64,
50    /// Optional human-readable description of the event for logging and debugging.
51    pub description: Option<String>,
52}

54/// # Event Data
55///
56/// `EventData` is an enum that represents the various types of events that can occur
57/// in the Theater system, organized by subsystem. Each variant contains a specific
58/// event data structure from the corresponding subsystem.
59///
60/// ## Purpose
61///
62/// This enum provides a type-safe way to handle events from different subsystems
63/// within a single event chain. It allows for pattern matching on event types and
64/// encapsulates the domain-specific data for each type of event.
65///
66/// ## Example
67///
68/// ```rust
69/// use theater::events::{EventData, ChainEventData};
70/// use theater::events::http::HttpEventData;
71///
72/// // Process events based on their type
73/// fn handle_event(event: &ChainEventData) {
74///     match &event.data {
75///         EventData::Http(http_event) => {
76///             println!("Handling HTTP event");
77///             // Process HTTP-specific event data
78///         },
79///         EventData::Runtime(runtime_event) => {
80///             println!("Handling Runtime event");
81///             // Process Runtime-specific event data
82///         },
83///         // Handle other event types
84///         _ => println!("Other event type: {}", event.event_type),
85///     }
86/// }
87/// ```
88///
89/// ## Implementation Notes
90///
91/// New event types should be added as variants to this enum when new subsystems
92/// are implemented. Each variant should contain a properly defined data structure
93/// that represents the specific event type.
94#[derive(Debug, Clone, Serialize, Deserialize)]
95pub enum EventData {
96    /// Environment variable access events for reading host environment variables.
97    Environment(environment::EnvironmentEventData),
98    /// File system access events, such as reading or writing files.
99    Filesystem(filesystem::FilesystemEventData),
100    /// HTTP-related events, including requests, responses, and WebSocket interactions.
101    Http(http::HttpEventData),
102    /// Actor-to-actor messaging events for communication between actors.
103    Message(message::MessageEventData),
104    /// OS Process management events, such as spawning processes and I/O.
105    Process(process::ProcessEventData),
106    /// Random number generation events for generating random data.
107    Random(random::RandomEventData),
108    /// Runtime lifecycle events, such as initialization, state changes, and shutdown.
109    Runtime(runtime::RuntimeEventData),
110    /// Supervision events related to actor parent-child relationships.
111    Supervisor(supervisor::SupervisorEventData),
112    /// Content store access events for the key-value storage system.
113    Store(store::StoreEventData),
114    /// Timer and scheduling events for time-based operations.
115    Timing(timing::TimingEventData),
116    /// WebAssembly execution events related to the WASM VM.
117    Wasm(wasm::WasmEventData),
118    /// Theater runtime system events for the global runtime coordination.
119    TheaterRuntime(theater_runtime::TheaterRuntimeEventData),
120}

122impl ChainEventData {
123    /// Gets the event type identifier string.
124    ///
125    /// ## Purpose
126    ///
127    /// This method returns the event type string, which can be used for filtering,
128    /// routing, or categorizing events.
129    ///
130    /// ## Returns
131    ///
132    /// The event type as a String
133    ///
134    /// ## Example
135    ///
136    /// ```rust
137    /// # use theater::events::ChainEventData;
138    /// # use theater::events::EventData;
139    /// # use theater::events::runtime::RuntimeEventData;
140    /// # let event = ChainEventData {
141    /// #     event_type: "runtime.init".to_string(),
142    /// #     data: EventData::Runtime(RuntimeEventData::InitCall { params: String::new() }),
143    /// #     timestamp: 0,
144    /// #     description: None,
145    /// # };
146    ///
147    /// let event_type = event.event_type();
148    /// println!("Event type: {}", event_type);
149    ///
150    /// // Filter events based on type
151    /// if event_type.starts_with("runtime.") {
152    ///     println!("This is a runtime event");
153    /// }
154    /// ```
155    #[allow(dead_code)]
156    pub fn event_type(&self) -> String {
157        let event_type = self.event_type.clone();
158        event_type
159    }
160
161    /// Gets the human-readable description of the event, if available.
162    ///
163    /// ## Purpose
164    ///
165    /// This method returns the optional human-readable description of the event,
166    /// which can be used for logging, debugging, or displaying to users. If no
167    /// description is available, it returns an empty string.
168    ///
169    /// ## Returns
170    ///
171    /// The event description as a String, or an empty string if none is available
172    ///
173    /// ## Example
174    ///
175    /// ```rust
176    /// # use theater::events::ChainEventData;
177    /// # use theater::events::EventData;
178    /// # use theater::events::runtime::RuntimeEventData;
179    /// # let event = ChainEventData {
180    /// #     event_type: "runtime.init".to_string(),
181    /// #     data: EventData::Runtime(RuntimeEventData::InitCall { params: String::new() }),
182    /// #     timestamp: 0,
183    /// #     description: Some("Actor initialized".to_string()),
184    /// # };
185    ///
186    /// let description = event.description();
187    /// println!("Event description: {}", description);
188    /// ```
189    #[allow(dead_code)]
190    pub fn description(&self) -> String {
191        match &self.description {
192            Some(desc) => desc.clone(),
193            None => String::from(""),
194        }
195    }
196
197    /// Serializes the event data to JSON.
198    ///
199    /// ## Purpose
200    ///
201    /// This method converts the event data to a JSON byte array, which can be used
202    /// for storage, transmission over the network, or inclusion in a chain event.
203    ///
204    /// ## Returns
205    ///
206    /// * `Ok(Vec<u8>)` - The serialized JSON data
207    /// * `Err(serde_json::Error)` - If serialization fails
208    ///
209    /// ## Example
210    ///
211    /// ```rust
212    /// # use theater::events::ChainEventData;
213    /// # use theater::events::EventData;
214    /// # use theater::events::runtime::RuntimeEventData;
215    /// # let event = ChainEventData {
216    /// #     event_type: "runtime.init".to_string(),
217    /// #     data: EventData::Runtime(RuntimeEventData::InitCall { params: String::new() }),
218    /// #     timestamp: 0,
219    /// #     description: None,
220    /// # };
221    ///
222    /// match event.to_json() {
223    ///     Ok(json_bytes) => {
224    ///         println!("Serialized event: {} bytes", json_bytes.len());
225    ///         // Store or transmit the JSON
226    ///     },
227    ///     Err(e) => println!("Failed to serialize event: {}", e),
228    /// }
229    /// ```
230    #[allow(dead_code)]
231    pub fn to_json(&self) -> Result<Vec<u8>, serde_json::Error> {
232        serde_json::to_vec(self)
233    }
234
235    /// Converts the typed event data to a generic chain event.
236    ///
237    /// ## Purpose
238    ///
239    /// This method creates a `ChainEvent` from the typed event data, which is used
240    /// to add the event to an actor's event chain. It serializes the event data and
241    /// includes it in the chain event, along with metadata like the event type and
242    /// timestamp.
243    ///
244    /// ## Parameters
245    ///
246    /// * `parent_hash` - Optional hash of the parent event in the chain
247    ///
248    /// ## Returns
249    ///
250    /// A new `ChainEvent` with the serialized event data and metadata
251    ///
252    /// ## Example
253    ///
254    /// ```rust
255    /// # use theater::events::ChainEventData;
256    /// # use theater::events::EventData;
257    /// # use theater::events::runtime::RuntimeEventData;
258    /// # use theater::chain::ChainEvent;
259    /// # let event_data = ChainEventData {
260    /// #     event_type: "runtime.init".to_string(),
261    /// #     data: EventData::Runtime(RuntimeEventData::InitCall { params: String::new() }),
262    /// #     timestamp: 0,
263    /// #     description: None,
264    /// # };
265    ///
266    /// // Convert to a chain event with no parent (root event)
267    /// let chain_event = event_data.to_chain_event(None);
268    ///
269    /// // Later, create child events in the chain
270    /// // Create a child event
271/// let child_event_data = ChainEventData {
272///     event_type: "child.event".to_string(),
273///     data: EventData::Runtime(RuntimeEventData::Log { level: "info".to_string(), message: "child event".to_string() }),
274///     timestamp: 0,
275///     description: None,
276/// };
277    /// let child_chain_event = child_event_data.to_chain_event(Some(chain_event.hash.clone()));
278    /// ```
279    ///
280    /// ## Implementation Notes
281    ///
282    /// The resulting `ChainEvent` will have an empty hash, which is filled in by the
283    /// `StateChain` when the event is added to the chain. The hash is calculated based
284    /// on the event content and cannot be known until the event is fully formed.
285    pub fn to_chain_event(&self, parent_hash: Option<Vec<u8>>) -> ChainEvent {
286        ChainEvent {
287            parent_hash,
288            hash: vec![],
289            event_type: self.event_type.clone(),
290            data: serde_json::to_vec(&self.data).unwrap_or_else(|_| vec![]),
291            timestamp: self.timestamp,
292            description: self.description.clone(),
293        }
294    }
295}
