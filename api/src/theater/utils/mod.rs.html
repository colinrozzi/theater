14/// Resolve a reference to a byte array.
15/// A reference can be a file path, a store path, or in the future a URL.
16/// The reference is resolved to a byte array.
17///
18/// a store path is a reference to a file in the store and is prefixed with `store://`
19/// a store path can be either a hash or a path to a label.
20/// store://<store-id>/hash/<hash>
21/// store://<store-id>/<label>
22///
23/// everything else is treated as a file path.
24pub async fn resolve_reference(reference: &str) -> Result<Vec<u8>, ReferenceError> {
25    info!("Resolving reference: {}", reference);
26
27    if reference.starts_with("store://") {
28        // store path
29        let parts: Vec<&str> = reference.split('/').collect();
30        if parts.len() < 3 {
31            return Err(ReferenceError::ResolveError(format!(
32                "Invalid store path: {}",
33                reference
34            )));
35        }
36        let store_id = parts[2];
37        let store = ContentStore::from_id(store_id);
38        if parts.len() == 4 && parts[3] == "hash" {
39            // store path with hash
40            let hash = parts[4];
41            let content_ref = ContentRef::from_str(hash);
42            info!("Resolving store path with hash: {}", hash);
43            store
44                .get(&content_ref)
45                .await
46                .map_err(|e| ReferenceError::ResolveError(e.to_string()))
47        } else {
48            // store path with label
49            let label = parts[3];
50            info!("Resolving store path with label: {}", label);
51            match store.get_content_by_label(label).await {
52                Ok(result) => match result {
53                    Some(content) => Ok(content),
54                    None => Err(ReferenceError::ResolveError(format!(
55                        "Label not found: {}",
56                        label
57                    ))),
58                },
59                Err(e) => Err(ReferenceError::ResolveError(e.to_string())),
60            }
61        }
62    } else {
63        // file path
64        tokio::fs::read(reference)
65            .await
66            .map_err(|e| ReferenceError::ResolveError(e.to_string()))
67    }
68}

70/// Merge two optional initial states, with the override state taking precedence
71/// for any overlapping keys.
72///
73/// The states should be JSON objects. If either state is not a valid JSON object,
74/// the following rules apply:
75/// - If config_state is None and override_state is None, returns None
76/// - If config_state is Some and override_state is None, returns config_state
77/// - If config_state is None and override_state is Some, returns override_state
78/// - If both are Some but either is not a valid JSON object, returns override_state
79/// - If both are Some and both are valid JSON objects, merges them with override_state taking precedence
80pub fn merge_initial_states(
81    config_state: Option<Vec<u8>>,
82    override_state: Option<Vec<u8>>,
83) -> Result<Option<Vec<u8>>> {
84    match (config_state, override_state) {
85        (None, None) => Ok(None),
86        (Some(state), None) => Ok(Some(state)),
87        (None, Some(state)) => Ok(Some(state)),
88        (Some(config_state), Some(override_state)) => {
89            // Parse both states
90            let config_json_result = serde_json::from_slice(&config_state);
91            let override_json_result = serde_json::from_slice(&override_state);
92
93            match (config_json_result, override_json_result) {
94                (Ok(mut config_json), Ok(override_json)) => {
95                    // Ensure both are objects
96                    if let (
97                        serde_json::Value::Object(ref mut config_map),
98                        serde_json::Value::Object(override_map),
99                    ) = (&mut config_json, &override_json)
100                    {
101                        // Merge override values into config
102                        for (key, value) in override_map {
103                            config_map.insert(key.clone(), value.clone());
104                        }
105
106                        // Serialize the merged result
107                        Ok(Some(serde_json::to_vec(&config_json)?))
108                    } else {
109                        // If either isn't an object, just use the override
110                        info!("Either initial state is not a JSON object, using override state");
111                        Ok(Some(override_state))
112                    }
113                }
114                _ => {
115                    // If either parsing fails, just use the override
116                    info!(
117                        "Failed to parse one of the initial states as JSON, using override state"
118                    );
119                    Ok(Some(override_state))
120                }
121            }
122        }
123    }
124}
