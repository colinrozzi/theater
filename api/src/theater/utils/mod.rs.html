18/// Resolve a reference to a byte array.
19/// A reference can be a file path, a store path, or a URL.
20/// The reference is resolved to a byte array.
21///
22/// A store path is a reference to a file in the store and is prefixed with `store://`
23/// A store path can be either a hash or a path to a label:
24/// - store://<store-id>/hash/<hash>
25/// - store://<store-id>/<label>
26///
27/// A URL is any reference starting with `http://` or `https://`
28///
29/// Everything else is treated as a file path.
30pub async fn resolve_reference(reference: &str) -> Result<Vec<u8>, ReferenceError> {
31    info!("Resolving reference: {}", reference);
32
33    if reference.starts_with("store://") {
34        // store path
35        let parts: Vec<&str> = reference.split('/').collect();
36        if parts.len() < 3 {
37            return Err(ReferenceError::ResolveError(format!(
38                "Invalid store path: {}",
39                reference
40            )));
41        }
42        let store_id = parts[2];
43        let store = ContentStore::from_id(store_id);
44
45        if parts.len() >= 5 && parts[3] == "hash" {
46            // store path with hash
47            let hash = parts[4];
48            let content_ref = ContentRef::from_str(hash);
49            info!("Resolving store path with hash: {}", hash);
50            store
51                .get(&content_ref)
52                .await
53                .map_err(|e| ReferenceError::ResolveError(e.to_string()))
54        } else if parts.len() >= 4 {
55            // store path with label
56            let label_string = parts[3];
57            let label = Label::new(label_string.to_string());
58            info!("Resolving store path with label: {}", label);
59            match store.get_content_by_label(&label).await {
60                Ok(result) => match result {
61                    Some(content) => Ok(content),
62                    None => Err(ReferenceError::ResolveError(format!(
63                        "Label not found: {}",
64                        label
65                    ))),
66                },
67                Err(e) => Err(ReferenceError::ResolveError(e.to_string())),
68            }
69        } else {
70            return Err(ReferenceError::ResolveError(format!(
71                "Invalid store path format: {}",
72                reference
73            )));
74        }
75    } else if reference.starts_with("http://") || reference.starts_with("https://") {
76        // HTTP/HTTPS URL
77        info!("Fetching from URL: {}", reference);
78        let client = reqwest::Client::new();
79        match client.get(reference).send().await {
80            Ok(response) => {
81                if response.status().is_success() {
82                    match response.bytes().await {
83                        Ok(bytes) => Ok(bytes.to_vec()),
84                        Err(e) => Err(ReferenceError::ResolveError(format!(
85                            "Failed to read response body from {}: {}",
86                            reference, e
87                        ))),
88                    }
89                } else {
90                    Err(ReferenceError::ResolveError(format!(
91                        "HTTP request failed for {}: {} {}",
92                        reference,
93                        response.status().as_u16(),
94                        response
95                            .status()
96                            .canonical_reason()
97                            .unwrap_or("Unknown error")
98                    )))
99                }
100            }
101            Err(e) => Err(ReferenceError::ResolveError(format!(
102                "Failed to fetch from {}: {}",
103                reference, e
104            ))),
105        }
106    } else {
107        // file path
108        info!("Reading from file path: {}", reference);
109        tokio::fs::read(reference)
110            .await
111            .map_err(|e| ReferenceError::ResolveError(e.to_string()))
112    }
113}

115/// Merge two optional initial states, with the override state taking precedence
116/// for any overlapping keys.
117///
118/// The states should be JSON objects. If either state is not a valid JSON object,
119/// the following rules apply:
120/// - If config_state is None and override_state is None, returns None
121/// - If config_state is Some and override_state is None, returns config_state
122/// - If config_state is None and override_state is Some, returns override_state
123/// - If both are Some but either is not a valid JSON object, returns override_state
124/// - If both are Some and both are valid JSON objects, merges them with override_state taking precedence
125pub fn merge_initial_states(
126    config_state: Option<Vec<u8>>,
127    override_state: Option<Vec<u8>>,
128) -> Result<Option<Vec<u8>>> {
129    match (config_state, override_state) {
130        (None, None) => Ok(None),
131        (Some(state), None) => Ok(Some(state)),
132        (None, Some(state)) => Ok(Some(state)),
133        (Some(config_state), Some(override_state)) => {
134            // Parse both states
135            let config_json_result = serde_json::from_slice(&config_state);
136            let override_json_result = serde_json::from_slice(&override_state);
137
138            match (config_json_result, override_json_result) {
139                (Ok(mut config_json), Ok(override_json)) => {
140                    // Ensure both are objects
141                    if let (
142                        serde_json::Value::Object(ref mut config_map),
143                        serde_json::Value::Object(override_map),
144                    ) = (&mut config_json, &override_json)
145                    {
146                        // Merge override values into config
147                        for (key, value) in override_map {
148                            config_map.insert(key.clone(), value.clone());
149                        }
150
151                        // Serialize the merged result
152                        Ok(Some(serde_json::to_vec(&config_json)?))
153                    } else {
154                        // If either isn't an object, just use the override
155                        info!("Either initial state is not a JSON object, using override state");
156                        Ok(Some(override_state))
157                    }
158                }
159                _ => {
160                    // If either parsing fails, just use the override
161                    info!(
162                        "Failed to parse one of the initial states as JSON, using override state"
163                    );
164                    Ok(Some(override_state))
165                }
166            }
167        }
168    }
169}

181/// Read events from filesystem for an actor that isn't currently running
182pub fn read_events_from_filesystem(actor_id: &TheaterId) -> Result<Vec<ChainEvent>> {
183    // Determine the Theater home directory
184    let theater_home = get_theater_home();
185
186    let chains_dir = format!("{}/chains", theater_home);
187    let events_dir = format!("{}/events", theater_home);
188
189    // Check if the actor's chain file exists
190    let chain_path = format!("{}/{}", chains_dir, actor_id);
191    if !std::path::Path::new(&chain_path).exists() {
192        debug!("No chain file found at: {}", chain_path);
193        return Err(anyhow!("No stored events found for actor: {}", actor_id));
194    }
195
196    // Read the chain head hash
197    let head_data = std::fs::read_to_string(&chain_path)?;
198    let head_hash: Option<Vec<u8>> = serde_json::from_str(&head_data)?;
199
200    if head_hash.is_none() {
201        debug!("Empty chain head for actor: {}", actor_id);
202        return Ok(Vec::new()); // Empty chain
203    }
204
205    // Reconstruct the full chain by following parent hash links
206    let mut events = Vec::new();
207    let mut current_hash = head_hash;
208
209    while let Some(hash) = current_hash {
210        let hash_hex = hex::encode(&hash);
211        let event_path = format!("{}/{}", events_dir, hash_hex);
212
213        // Read and parse the event
214        let event_data = match std::fs::read_to_string(&event_path) {
215            Ok(data) => data,
216            Err(e) => {
217                debug!("Failed to read event file {}: {}", event_path, e);
218                break; // Break the chain if we can't read an event file
219            }
220        };
221
222        let event = match serde_json::from_str::<ChainEvent>(&event_data) {
223            Ok(event) => event,
224            Err(e) => {
225                debug!("Failed to parse event from {}: {}", event_path, e);
226                break; // Break the chain if we can't parse an event
227            }
228        };
229
230        // Store the event and move to the parent
231        current_hash = event.parent_hash.clone();
232        events.push(event);
233    }
234
235    // Reverse the events to get them in chronological order (oldest first)
236    events.reverse();
237
238    debug!(
239        "Read {} events from filesystem for actor {}",
240        events.len(),
241        actor_id
242    );
243    Ok(events)
244}
