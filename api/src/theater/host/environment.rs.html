51    pub async fn setup_host_functions(&self, actor_component: &mut ActorComponent) -> Result<()> {
52        info!("Setting up environment host functions (read-only)");
53
54        let mut interface = actor_component
55            .linker
56            .instance("theater:simple/environment")
57            .expect("Could not instantiate theater:simple/environment");
58
59        let _config = self.config.clone();
60        let permissions = self.permissions.clone();
61
62        // get-var implementation
63        interface.func_wrap(
64            "get-var",
65            move |mut ctx: StoreContextMut<'_, ActorStore>,
66                  (var_name,): (String,)|
67                  -> Result<(Option<String>,)> {
68                let now = Utc::now().timestamp_millis() as u64;
69                
70                // PERMISSION CHECK BEFORE OPERATION
71                if let Err(e) = PermissionChecker::check_env_var_access(
72                    &permissions,
73                    &var_name,
74                ) {
75                    // Record permission denied event
76                    ctx.data_mut().record_event(ChainEventData {
77                        event_type: "theater:simple/environment/permission-denied".to_string(),
78                        data: EventData::Environment(EnvironmentEventData {
79                            operation: "get-var".to_string(),
80                            variable_name: var_name.clone(),
81                            success: false,
82                            value_found: false,
83                            timestamp: chrono::Utc::now(),
84                        }),
85                        timestamp: now,
86                        description: Some(format!(
87                            "Permission denied for environment variable access: {}",
88                            e
89                        )),
90                    });
91                    return Ok((None,));
92                }
93
94                let value = env::var(&var_name).ok();
95                let value_found = value.is_some();
96
97                // Record the access attempt
98                ctx.data_mut().record_event(ChainEventData {
99                    event_type: "theater:simple/environment/get-var".to_string(),
100                    data: EventData::Environment(EnvironmentEventData {
101                        operation: "get-var".to_string(),
102                        variable_name: var_name.clone(),
103                        success: true,
104                        value_found,
105                        timestamp: chrono::Utc::now(),
106                    }),
107                    timestamp: now,
108                    description: Some(format!(
109                        "Environment variable access: {} (found: {})",
110                        var_name, value_found
111                    )),
112                });
113
114                Ok((value,))
115            },
116        )?;
117
118        let permissions_clone = self.permissions.clone();
119
120        // exists implementation
121        interface.func_wrap(
122            "exists",
123            move |mut ctx: StoreContextMut<'_, ActorStore>,
124                  (var_name,): (String,)|
125                  -> Result<(bool,)> {
126                let now = Utc::now().timestamp_millis() as u64;
127                
128                // PERMISSION CHECK BEFORE OPERATION
129                if let Err(e) = PermissionChecker::check_env_var_access(
130                    &permissions_clone,
131                    &var_name,
132                ) {
133                    // Record permission denied event
134                    ctx.data_mut().record_event(ChainEventData {
135                        event_type: "theater:simple/environment/permission-denied".to_string(),
136                        data: EventData::Environment(EnvironmentEventData {
137                            operation: "exists".to_string(),
138                            variable_name: var_name.clone(),
139                            success: false,
140                            value_found: false,
141                            timestamp: chrono::Utc::now(),
142                        }),
143                        timestamp: now,
144                        description: Some(format!(
145                            "Permission denied for environment variable exists check: {}",
146                            e
147                        )),
148                    });
149                    return Ok((false,));
150                }
151
152                let exists = env::var(&var_name).is_ok();
153
154                // Record the check
155                ctx.data_mut().record_event(ChainEventData {
156                    event_type: "theater:simple/environment/exists".to_string(),
157                    data: EventData::Environment(EnvironmentEventData {
158                        operation: "exists".to_string(),
159                        variable_name: var_name.clone(),
160                        success: true,
161                        value_found: exists,
162                        timestamp: chrono::Utc::now(),
163                    }),
164                    timestamp: now,
165                    description: Some(format!(
166                        "Environment variable exists check: {} (exists: {})",
167                        var_name, exists
168                    )),
169                });
170
171                Ok((exists,))
172            },
173        )?;
174
175        let config_clone2 = self.config.clone();
176
177        // list-vars implementation
178        interface.func_wrap(
179            "list-vars",
180            move |mut ctx: StoreContextMut<'_, ActorStore>,
181                  (): ()|
182                  -> Result<(Vec<(String, String)>,)> {
183                let now = Utc::now().timestamp_millis() as u64;
184
185                if !config_clone2.allow_list_all {
186                    // Record denied list attempt
187                    ctx.data_mut().record_event(ChainEventData {
188                        event_type: "theater:simple/environment/list-vars".to_string(),
189                        data: EventData::Environment(EnvironmentEventData {
190                            operation: "list-vars".to_string(),
191                            variable_name: "(list-all disabled)".to_string(),
192                            success: false,
193                            value_found: false,
194                            timestamp: chrono::Utc::now(),
195                        }),
196                        timestamp: now,
197                        description: Some(
198                            "Environment variable listing denied - allow_list_all is false"
199                                .to_string(),
200                        ),
201                    });
202                    return Ok((Vec::new(),));
203                }
204
205                let mut accessible_vars = Vec::new();
206
207                for (key, value) in env::vars() {
208                    if config_clone2.is_variable_allowed(&key) {
209                        accessible_vars.push((key, value));
210                    }
211                }
212
213                // Record the list operation
214                let count = accessible_vars.len();
215                ctx.data_mut().record_event(ChainEventData {
216                    event_type: "theater:simple/environment/list-vars".to_string(),
217                    data: EventData::Environment(EnvironmentEventData {
218                        operation: "list-vars".to_string(),
219                        variable_name: format!("(returned {} variables)", count),
220                        success: true,
221                        value_found: count > 0,
222                        timestamp: chrono::Utc::now(),
223                    }),
224                    timestamp: now,
225                    description: Some(format!(
226                        "Environment variable listing returned {} accessible variables",
227                        count
228                    )),
229                });
230
231                Ok((accessible_vars,))
232            },
233        )?;
234
235        Ok(())
236    }
