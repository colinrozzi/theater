51    pub async fn setup_host_functions(&self, actor_component: &mut ActorComponent) -> Result<()> {
52        info!("Setting up environment host functions (read-only)");
53
54        let mut interface = actor_component
55            .linker
56            .instance("ntwk:theater/environment")
57            .expect("Could not instantiate ntwk:theater/environment");
58
59        let config = self.config.clone();
60        
61        // get-var implementation
62        interface.func_wrap(
63            "get-var",
64            move |mut ctx: StoreContextMut<'_, ActorStore>, (var_name,): (String,)| -> Result<(Option<String>,)> {
65                let now = Utc::now().timestamp_millis() as u64;
66                let is_allowed = config.is_variable_allowed(&var_name);
67                
68                if !is_allowed {
69                    // Record access denial
70                    ctx.data_mut().record_event(ChainEventData {
71                        event_type: "ntwk:theater/environment/get-var".to_string(),
72                        data: EventData::Environment(EnvironmentEventData {
73                            operation: "get-var".to_string(),
74                            variable_name: var_name.clone(),
75                            success: false,
76                            value_found: false,
77                            timestamp: chrono::Utc::now(),
78                        }),
79                        timestamp: now,
80                        description: Some(format!("Access denied for environment variable: {}", var_name)),
81                    });
82                    return Ok((None,));
83                }
84
85                let value = env::var(&var_name).ok();
86                let value_found = value.is_some();
87
88                // Record the access attempt
89                ctx.data_mut().record_event(ChainEventData {
90                    event_type: "ntwk:theater/environment/get-var".to_string(),
91                    data: EventData::Environment(EnvironmentEventData {
92                        operation: "get-var".to_string(),
93                        variable_name: var_name.clone(),
94                        success: true,
95                        value_found,
96                        timestamp: chrono::Utc::now(),
97                    }),
98                    timestamp: now,
99                    description: Some(format!("Environment variable access: {} (found: {})", var_name, value_found)),
100                });
101
102                Ok((value,))
103            },
104        )?;
105
106        let config_clone = self.config.clone();
107
108        // exists implementation
109        interface.func_wrap(
110            "exists",
111            move |mut ctx: StoreContextMut<'_, ActorStore>, (var_name,): (String,)| -> Result<(bool,)> {
112                let now = Utc::now().timestamp_millis() as u64;
113                let is_allowed = config_clone.is_variable_allowed(&var_name);
114                
115                if !is_allowed {
116                    // Record access denial
117                    ctx.data_mut().record_event(ChainEventData {
118                        event_type: "ntwk:theater/environment/exists".to_string(),
119                        data: EventData::Environment(EnvironmentEventData {
120                            operation: "exists".to_string(),
121                            variable_name: var_name.clone(),
122                            success: false,
123                            value_found: false,
124                            timestamp: chrono::Utc::now(),
125                        }),
126                        timestamp: now,
127                        description: Some(format!("Access denied for environment variable exists check: {}", var_name)),
128                    });
129                    return Ok((false,));
130                }
131
132                let exists = env::var(&var_name).is_ok();
133
134                // Record the check
135                ctx.data_mut().record_event(ChainEventData {
136                    event_type: "ntwk:theater/environment/exists".to_string(),
137                    data: EventData::Environment(EnvironmentEventData {
138                        operation: "exists".to_string(),
139                        variable_name: var_name.clone(),
140                        success: true,
141                        value_found: exists,
142                        timestamp: chrono::Utc::now(),
143                    }),
144                    timestamp: now,
145                    description: Some(format!("Environment variable exists check: {} (exists: {})", var_name, exists)),
146                });
147
148                Ok((exists,))
149            },
150        )?;
151
152        let config_clone2 = self.config.clone();
153
154        // list-vars implementation
155        interface.func_wrap(
156            "list-vars",
157            move |mut ctx: StoreContextMut<'_, ActorStore>, (): ()| -> Result<(Vec<(String, String)>,)> {
158                let now = Utc::now().timestamp_millis() as u64;
159                
160                if !config_clone2.allow_list_all {
161                    // Record denied list attempt
162                    ctx.data_mut().record_event(ChainEventData {
163                        event_type: "ntwk:theater/environment/list-vars".to_string(),
164                        data: EventData::Environment(EnvironmentEventData {
165                            operation: "list-vars".to_string(),
166                            variable_name: "(list-all disabled)".to_string(),
167                            success: false,
168                            value_found: false,
169                            timestamp: chrono::Utc::now(),
170                        }),
171                        timestamp: now,
172                        description: Some("Environment variable listing denied - allow_list_all is false".to_string()),
173                    });
174                    return Ok((Vec::new(),));
175                }
176
177                let mut accessible_vars = Vec::new();
178                
179                for (key, value) in env::vars() {
180                    if config_clone2.is_variable_allowed(&key) {
181                        accessible_vars.push((key, value));
182                    }
183                }
184
185                // Record the list operation
186                let count = accessible_vars.len();
187                ctx.data_mut().record_event(ChainEventData {
188                    event_type: "ntwk:theater/environment/list-vars".to_string(),
189                    data: EventData::Environment(EnvironmentEventData {
190                        operation: "list-vars".to_string(),
191                        variable_name: format!("(returned {} variables)", count),
192                        success: true,
193                        value_found: count > 0,
194                        timestamp: chrono::Utc::now(),
195                    }),
196                    timestamp: now,
197                    description: Some(format!("Environment variable listing returned {} accessible variables", count)),
198                });
199
200                Ok((accessible_vars,))
201            },
202        )?;
203
204        Ok(())
205    }
