63    pub async fn setup_host_functions(&self, actor_component: &mut ActorComponent) -> Result<()> {
64        // Record setup start
65        actor_component.actor_store.record_event(ChainEventData {
66            event_type: "http-client-setup".to_string(),
67            data: EventData::Http(HttpEventData::HandlerSetupStart),
68            timestamp: chrono::Utc::now().timestamp_millis() as u64,
69            description: Some("Starting HTTP client host function setup".to_string()),
70        });
71
72        info!("Setting up http client host functions");
73
74        let mut interface = match actor_component
75            .linker
76            .instance("theater:simple/http-client")
77        {
78            Ok(interface) => {
79                // Record successful linker instance creation
80                actor_component.actor_store.record_event(ChainEventData {
81                    event_type: "http-client-setup".to_string(),
82                    data: EventData::Http(HttpEventData::LinkerInstanceSuccess),
83                    timestamp: chrono::Utc::now().timestamp_millis() as u64,
84                    description: Some("Successfully created linker instance".to_string()),
85                });
86                interface
87            }
88            Err(e) => {
89                // Record the specific error where it happens
90                actor_component.actor_store.record_event(ChainEventData {
91                    event_type: "http-client-setup".to_string(),
92                    data: EventData::Http(HttpEventData::HandlerSetupError {
93                        error: e.to_string(),
94                        step: "linker_instance".to_string(),
95                    }),
96                    timestamp: chrono::Utc::now().timestamp_millis() as u64,
97                    description: Some(format!("Failed to create linker instance: {}", e)),
98                });
99                return Err(anyhow::anyhow!("Could not instantiate theater:simple/http-client: {}", e));
100            }
101        };
102
103        let permissions = self.permissions.clone();
104        interface.func_wrap_async(
105            "send-http",
106            move |mut ctx: wasmtime::StoreContextMut<'_, ActorStore>,
107                  (req,): (HttpRequest,)|
108                  -> Box<dyn Future<Output = Result<(Result<HttpResponse, String>,)>> + Send> {
109                
110                // Record HTTP client request call event
111                ctx.data_mut().record_event(ChainEventData {
112                    event_type: "theater:simple/http-client/send-http".to_string(),
113                    data: EventData::Http(HttpEventData::HttpClientRequestCall {
114                        method: req.method.clone(),
115                        url: req.uri.clone(),
116                        headers_count: req.headers.len(),
117                        body: req.body.clone().map(|b| String::from_utf8_lossy(&b).to_string()),
118                    }),
119                    timestamp: chrono::Utc::now().timestamp_millis() as u64,
120                    description: Some(format!("Sending {} request to {}", req.method, req.uri)),
121                });
122                
123                // PERMISSION CHECK BEFORE OPERATION
124                // Extract host from URL for permission checking
125                let host = if let Ok(parsed_url) = reqwest::Url::parse(&req.uri) {
126                    parsed_url.host_str().unwrap_or(&req.uri).to_string()
127                } else {
128                    req.uri.clone()
129                };
130                
131                if let Err(e) = PermissionChecker::check_http_operation(
132                    &permissions,
133                    &req.method,
134                    &host,
135                ) {
136                    error!("HTTP client permission denied: {}", e);
137                    ctx.data_mut().record_event(ChainEventData {
138                        event_type: "theater:simple/http-client/permission-denied".to_string(),
139                        data: EventData::Http(HttpEventData::PermissionDenied {
140                            operation: "send-http".to_string(),
141                            method: req.method.clone(),
142                            url: req.uri.clone(),
143                            reason: e.to_string(),
144                        }),
145                        timestamp: chrono::Utc::now().timestamp_millis() as u64,
146                        description: Some(format!("Permission denied for {} request to {}", req.method, req.uri)),
147                    });
148                    return Box::new(async move {
149                        Ok((Err(format!("Permission denied: {}", e)),))
150                    });
151                }
152                
153                let req_clone = req.clone();
154                
155                Box::new(async move {
156                    let client = reqwest::Client::new();
157                    
158                    // Parse method or return error
159                    let method = match Method::from_bytes(req_clone.method.as_bytes()) {
160                        Ok(m) => m,
161                        Err(e) => {
162                            let err_msg = format!("Invalid HTTP method: {}", e);
163                            
164                            // Record error event
165                            ctx.data_mut().record_event(ChainEventData {
166                                event_type: "theater:simple/http-client/send-http".to_string(),
167                                data: EventData::Http(HttpEventData::Error {
168                                    operation: "send-http".to_string(),
169                                    path: req_clone.uri.clone(),
170                                    message: err_msg.clone(),
171                                }),
172                                timestamp: chrono::Utc::now().timestamp_millis() as u64,
173                                description: Some(format!("Error sending request to {}: {}", req_clone.uri, err_msg)),
174                            });
175                            
176                            return Ok((Err(err_msg),));
177                        }
178                    };
179                    
180                    let mut request = client.request(method, req_clone.uri.clone());
181                    
182                    for (key, value) in req_clone.headers {
183                        request = request.header(key, value);
184                    }
185                    if let Some(body) = req_clone.body {
186                        request = request.body(body);
187                    }
188                    
189                    info!("Sending {} request to {}", req_clone.method, req_clone.uri);
190
191                    match request.send().await {
192                        Ok(response) => {
193                            let status = response.status().as_u16();
194                            let headers = response
195                                .headers()
196                                .iter()
197                                .map(|(key, value)| {
198                                    (
199                                        key.as_str().to_string(),
200                                        value.to_str().unwrap_or_default().to_string(),
201                                    )
202                                })
203                                .collect();
204                            
205                            let body = match response.bytes().await {
206                                Ok(bytes) => Some(bytes.to_vec()),
207                                Err(e) => {
208                                    // Record error reading response body
209                                    ctx.data_mut().record_event(ChainEventData {
210                                        event_type: "theater:simple/http-client/send-http".to_string(),
211                                        data: EventData::Http(HttpEventData::Error {
212                                            operation: "read-response-body".to_string(),
213                                            path: req_clone.uri.clone(),
214                                            message: e.to_string(),
215                                        }),
216                                        timestamp: chrono::Utc::now().timestamp_millis() as u64,
217                                        description: Some(format!("Error reading response body from {}: {}", req_clone.uri, e)),
218                                    });
219                                    
220                                    None
221                                }
222                            };
223                            
224                            let resp = HttpResponse {
225                                status,
226                                headers,
227                                body: body.clone(),
228                            };
229                            
230                            // Record HTTP client request result event
231                            ctx.data_mut().record_event(ChainEventData {
232                                event_type: "theater:simple/http-client/send-http".to_string(),
233                                data: EventData::Http(HttpEventData::HttpClientRequestResult {
234                                    status,
235                                    headers_count: resp.headers.len(),
236                                    body: body.clone().map(|b| String::from_utf8_lossy(&b).to_string()),
237                                    success: true,
238                                }),
239                                timestamp: chrono::Utc::now().timestamp_millis() as u64,
240                                description: Some(format!("Received response from {} with status {}", req_clone.uri, status)),
241                            });
242                            
243                            Ok((Ok(resp),))
244                        }
245                        Err(e) => {
246                            let err_msg = e.to_string();
247                            
248                            // Record HTTP client request error event
249                            ctx.data_mut().record_event(ChainEventData {
250                                event_type: "theater:simple/http-client/send-http".to_string(),
251                                data: EventData::Http(HttpEventData::Error {
252                                    operation: "send-http".to_string(),
253                                    path: req_clone.uri.clone(),
254                                    message: err_msg.clone(),
255                                }),
256                                timestamp: chrono::Utc::now().timestamp_millis() as u64,
257                                description: Some(format!("Error sending request to {}: {}", req_clone.uri, err_msg)),
258                            });
259                            
260                            Ok((Err(err_msg),))
261                        }
262                    }
263                })
264            },
265        )?;
266
267        // Record overall setup completion
268        actor_component.actor_store.record_event(ChainEventData {
269            event_type: "http-client-setup".to_string(),
270            data: EventData::Http(HttpEventData::HandlerSetupSuccess),
271            timestamp: chrono::Utc::now().timestamp_millis() as u64,
272            description: Some("HTTP client host functions setup completed successfully".to_string()),
273        });
274
275        info!("Host functions set up for http-client");
276
277        Ok(())
278    }
