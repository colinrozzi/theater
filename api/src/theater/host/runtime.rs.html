73    pub async fn setup_host_functions(&self, actor_component: &mut ActorComponent) -> Result<()> {
74        info!("Setting up runtime host functions");
75        let name = actor_component.name.clone();
76        let mut interface = actor_component
77            .linker
78            .instance("ntwk:theater/runtime")
79            .expect("Could not instantiate ntwk:theater/runtime");
80
81        interface
82            .func_wrap(
83                "log",
84                move |mut ctx: wasmtime::StoreContextMut<'_, ActorStore>, (msg,): (String,)| {
85                    let id = ctx.data().id.clone();
86
87                    // Record log call event
88                    ctx.data_mut().record_event(ChainEventData {
89                        event_type: "ntwk:theater/runtime/log".to_string(),
90                        data: EventData::Runtime(RuntimeEventData::Log {
91                            level: "info".to_string(),
92                            message: msg.clone(),
93                        }),
94                        timestamp: chrono::Utc::now().timestamp_millis() as u64,
95                        description: Some(format!("Actor log: {}", msg)),
96                    });
97
98                    info!("[ACTOR] [{}] [{}] {}", id, name, msg);
99                    Ok(())
100                },
101            )
102            .expect("Failed to wrap log function");
103
104        interface
105            .func_wrap(
106                "get-state",
107                move |mut ctx: StoreContextMut<'_, ActorStore>, ()| -> Result<(Vec<u8>,)> {
108                    // Record state request call event
109                    ctx.data_mut().record_event(ChainEventData {
110                        event_type: "ntwk:theater/runtime/get-state".to_string(),
111                        data: EventData::Runtime(RuntimeEventData::StateChangeCall {
112                            old_state: "unknown".to_string(),
113                            new_state: "requested".to_string(),
114                        }),
115                        timestamp: chrono::Utc::now().timestamp_millis() as u64,
116                        description: Some("Get state request".to_string()),
117                    });
118
119                    // Return current state
120                    let state = ctx
121                        .data()
122                        .get_last_event()
123                        .map(|e| e.data.clone())
124                        .unwrap_or_default();
125
126                    // Record state request result event
127                    ctx.data_mut().record_event(ChainEventData {
128                        event_type: "ntwk:theater/runtime/get-state".to_string(),
129                        data: EventData::Runtime(RuntimeEventData::StateChangeResult {
130                            success: true,
131                        }),
132                        timestamp: chrono::Utc::now().timestamp_millis() as u64,
133                        description: Some(format!("State retrieved: {} bytes", state.len())),
134                    });
135
136                    Ok((state,))
137                },
138            )
139            .expect("Failed to wrap get-state function");
140
141        let name = actor_component.name.clone();
142        let theater_tx = self.theater_tx.clone();
143
144        interface
145            .func_wrap_async(
146                "shutdown",
147                move |mut ctx: wasmtime::StoreContextMut<'_, ActorStore>, (data,): (Option<Vec<u8>>,)|
148                      -> Box<dyn Future<Output = Result<(Result<(), String>,)>> + Send> {
149                    // Record shutdown call event
150                    ctx.data_mut().record_event(ChainEventData {
151                        event_type: "ntwk:theater/runtime/shutdown".to_string(),
152                        data: EventData::Runtime(RuntimeEventData::ShutdownCall {
153                            data: data.clone(),
154                        }),
155                        timestamp: chrono::Utc::now().timestamp_millis() as u64,
156                        description: Some(format!("Actor shutdown with data: {:?}", data)),
157                    });
158
159                    info!(
160                        "[ACTOR] [{}] [{}] Shutdown requested: {:?}",
161                        ctx.data().id,
162                        name,
163                        data
164                    );
165                    let theater_tx = theater_tx.clone();
166
167                    Box::new(async move {
168                        match theater_tx
169                            .send(TheaterCommand::ShuttingDown {
170                                actor_id: ctx.data().id.clone(),
171                                data,
172                            })
173                            .await
174                        {
175                            Ok(_) => {
176                                ctx.data_mut().record_event(ChainEventData {
177                                    event_type: "ntwk:theater/runtime/shutdown".to_string(),
178                                    data: EventData::Runtime(RuntimeEventData::ShutdownRequested {
179                                        success: true,
180                                    }),
181                                    timestamp: chrono::Utc::now().timestamp_millis() as u64,
182                                    description: Some("Shutdown command sent successfully".to_string()),
183                                });
184                                Ok((Ok(()),))
185                            }
186                            Err(e) => {
187                                let err = e.to_string();
188                                // Record failed shutdown result event
189                                ctx.data_mut().record_event(ChainEventData {
190                                    event_type: "ntwk:theater/runtime/shutdown".to_string(),
191                                    data: EventData::Runtime(RuntimeEventData::ShutdownRequested {
192                                        success: false,
193                                    }),
194                                    timestamp: chrono::Utc::now().timestamp_millis() as u64,
195                                    description: Some(format!(
196                                        "Failed to send shutdown command: {}",
197                                        err
198                                    )),
199                                });
200                                Ok((Err(err),))
201                            }
202                        }
203                    })
204                },
205            )
206            .expect("Failed to wrap shutdown function");
207
208        Ok(())
209    }
