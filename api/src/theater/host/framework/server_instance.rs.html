234    // NEW: Completely rewritten router building using Axum's native routing
235    fn build_router(&self, actor_handle: ActorHandle) -> Router {
236        let mut router = Router::new();
237        
238        // Create shared base state
239        let base_state = Arc::new(ServerState {
240            id: self.id,
241            actor_handle,
242            middlewares: self.middlewares.clone(),
243            websockets: self.websockets.clone(),
244            active_ws_connections: self.active_ws_connections.clone(),
245        });
246
247        // Add each route individually with Axum's native method routing
248        for route in self.routes.values() {
249            let route_state = RouteHandlerState {
250                handler_id: route.handler_id,
251                server_state: base_state.clone(),
252            };
253
254            debug!("Adding route: {} {} -> handler {}", route.method, route.path, route.handler_id);
255
256            // Use the appropriate HTTP method handler - Axum handles all the wildcard magic!
257            let method_handler = match route.method.as_str() {
258                "GET" => get(Self::handle_request),
259                "POST" => post(Self::handle_request),
260                "PUT" => put(Self::handle_request),
261                "DELETE" => delete(Self::handle_request),
262                "PATCH" => patch(Self::handle_request),
263                "HEAD" => head(Self::handle_request),
264                "OPTIONS" => options(Self::handle_request),
265                "*" => any(Self::handle_request),
266                _ => {
267                    warn!("Unknown HTTP method: {}, using any()", route.method);
268                    any(Self::handle_request)
269                }
270            };
271
272            // Let Axum handle the path pattern matching - this is where the magic happens!
273            router = router.route(&route.path, method_handler.with_state(route_state));
274        }
275
276        // Add WebSocket routes
277        for (ws_path, ws_config) in &self.websockets {
278            let ws_state = WebSocketHandlerState {
279                config: ws_config.clone(),
280                server_state: base_state.clone(),
281            };
282            
283            debug!("Adding WebSocket route: {}", ws_path);
284            router = router.route(
285                ws_path,
286                get(Self::handle_websocket_upgrade).with_state(ws_state),
287            );
288        }
289
290        info!("Built router with {} routes and {} WebSocket endpoints", self.routes.len(), self.websockets.len());
291        router
292    }

295    async fn handle_request(
296        State(route_state): State<RouteHandlerState>,
297        matched_path: MatchedPath,
298        // Axum automatically extracts path parameters based on the route pattern
299        path_params: Option<Path<HashMap<String, String>>>,
300        req: Request<axum::body::Body>,
301    ) -> Response<axum::body::Body> {
302        
303        let actual_path = req.uri().path().to_string();
304        let method = req.method().as_str().to_uppercase();
305        
306        debug!("Handling {} request to {} (matched pattern: {})", method, actual_path, matched_path.as_str());
307        
308        // Convert request, including path parameters extracted by Axum
309        let mut theater_request = convert_request_with_axum_params(req, path_params).await;
310        
311        // Apply middlewares (using the actual requested path, not the pattern)
312        let middlewares = route_state.server_state.find_middlewares(&actual_path);
313        for middleware in middlewares {
314            match route_state.server_state.call_middleware(middleware.handler_id, theater_request.clone()).await {
315                Ok(result) => {
316                    if !result.proceed {
317                        debug!("Middleware {} rejected request", middleware.handler_id);
318                        return Response::builder()
319                            .status(StatusCode::FORBIDDEN)
320                            .body(axum::body::Body::empty())
321                            .unwrap_or_default();
322                    }
323                    theater_request = result.request;
324                }
325                Err(e) => {
326                    error!("Middleware error: {}", e);
327                    return Response::builder()
328                        .status(StatusCode::INTERNAL_SERVER_ERROR)
329                        .body(format!("Middleware error: {}", e).into())
330                        .unwrap_or_default();
331                }
332            }
333        }
334
335        // Call the specific handler for this route
336        match route_state.server_state.call_route_handler(route_state.handler_id, theater_request).await {
337            Ok(response) => {
338                debug!("Handler {} completed successfully", route_state.handler_id);
339                convert_response(response)
340            }
341            Err(e) => {
342                error!("Handler error for route {} ({}): {}", matched_path.as_str(), route_state.handler_id, e);
343                Response::builder()
344                    .status(StatusCode::INTERNAL_SERVER_ERROR)
345                    .body(format!("Handler error: {}", e).into())
346                    .unwrap_or_default()
347            }
348        }
349    }

364    async fn handle_websocket_connection(
365        ws_state: WebSocketHandlerState,
366        socket: axum::extract::ws::WebSocket,
367        path: String,
368    ) {
369        let connection_id = rand::random::<u64>();
370        let (mut sender, mut receiver) = socket.split();
371        let (tx, mut rx) = mpsc::channel::<WebSocketMessage>(100);
372
373        // Store connection
374        {
375            let mut connections = ws_state.server_state.active_ws_connections.write().await;
376            connections.insert(connection_id, WebSocketConnection {
377                id: connection_id,
378                sender: tx,
379            });
380        }
381
382        // Handle connect event
383        if let Some(connect_handler_id) = ws_state.config.connect_handler_id {
384            if let Err(e) = ws_state.server_state.actor_handle
385                .call_function::<(u64, u64, String, Option<String>), ()>(
386                    "theater:simple/http-handlers.handle-websocket-connect".to_string(),
387                    (connect_handler_id, connection_id, path.clone(), None),
388                )
389                .await
390            {
391                error!("WebSocket connect handler error: {}", e);
392            }
393        }
394
395        // Spawn task to send messages from channel to WebSocket
396        let send_task = tokio::spawn(async move {
397            while let Some(message) = rx.recv().await {
398                let ws_message = match message.ty {
399                    MessageType::Text => {
400                        if let Some(text) = message.text {
401                            axum::extract::ws::Message::Text(text.into())
402                        } else {
403                            continue;
404                        }
405                    }
406                    MessageType::Binary => {
407                        if let Some(data) = message.data {
408                            axum::extract::ws::Message::Binary(data.into())
409                        } else {
410                            continue;
411                        }
412                    }
413                    MessageType::Close => axum::extract::ws::Message::Close(None),
414                    MessageType::Ping => {
415                        if let Some(data) = message.data {
416                            axum::extract::ws::Message::Ping(data.into())
417                        } else {
418                            axum::extract::ws::Message::Ping(vec![].into())
419                        }
420                    }
421                    MessageType::Pong => {
422                        if let Some(data) = message.data {
423                            axum::extract::ws::Message::Pong(data.into())
424                        } else {
425                            axum::extract::ws::Message::Pong(vec![].into())
426                        }
427                    }
428                    _ => continue,
429                };
430
431                if sender.send(ws_message).await.is_err() {
432                    break;
433                }
434            }
435        });
436
437        // Handle incoming messages
438        while let Some(msg) = receiver.next().await {
439            match msg {
440                Ok(axum::extract::ws::Message::Text(text)) => {
441                    let message = WebSocketMessage {
442                        ty: MessageType::Text,
443                        data: None,
444                        text: Some(text.to_string()),
445                    };
446
447                    if let Err(e) = ws_state.server_state.actor_handle
448                        .call_function::<(u64, u64, WebSocketMessage), (Vec<WebSocketMessage>,)>(
449                            "theater:simple/http-handlers.handle-websocket-message".to_string(),
450                            (ws_state.config.message_handler_id, connection_id, message),
451                        )
452                        .await
453                    {
454                        error!("WebSocket message handler error: {}", e);
455                    }
456                }
457                Ok(axum::extract::ws::Message::Binary(data)) => {
458                    let message = WebSocketMessage {
459                        ty: MessageType::Binary,
460                        data: Some(data.to_vec()),
461                        text: None,
462                    };
463
464                    if let Err(e) = ws_state.server_state.actor_handle
465                        .call_function::<(u64, u64, WebSocketMessage), (Vec<WebSocketMessage>,)>(
466                            "theater:simple/http-handlers.handle-websocket-message".to_string(),
467                            (ws_state.config.message_handler_id, connection_id, message),
468                        )
469                        .await
470                    {
471                        error!("WebSocket message handler error: {}", e);
472                    }
473                }
474                Ok(axum::extract::ws::Message::Close(_)) => {
475                    break;
476                }
477                Err(e) => {
478                    error!("WebSocket error: {}", e);
479                    break;
480                }
481                _ => {}
482            }
483        }
484
485        // Clean up connection
486        {
487            let mut connections = ws_state.server_state.active_ws_connections.write().await;
488            connections.remove(&connection_id);
489        }
490
491        // Handle disconnect event
492        if let Some(disconnect_handler_id) = ws_state.config.disconnect_handler_id {
493            if let Err(e) = ws_state.server_state.actor_handle
494                .call_function::<(u64, u64), ()>(
495                    "theater:simple/http-handlers.handle-websocket-disconnect".to_string(),
496                    (disconnect_handler_id, connection_id),
497                )
498                .await
499            {
500                error!("WebSocket disconnect handler error: {}", e);
501            }
502        }
503
504        send_task.abort();
505    }

520    async fn start_http(&mut self, actor_handle: ActorHandle) -> Result<u16> {
521        let router = self.build_router_safe(actor_handle).map_err(|e| anyhow::anyhow!(e))?;
522        let host = self.config.host.clone().unwrap_or_else(|| "0.0.0.0".to_string());
523        
524        let addr = if self.port == 0 {
525            format!("{}:0", host)
526        } else {
527            format!("{}:{}", host, self.port)
528        };
529
530        let listener = tokio::net::TcpListener::bind(&addr).await?;
531        let actual_addr = listener.local_addr()?;
532        self.port = actual_addr.port();
533
534        let (shutdown_tx, shutdown_rx) = tokio::sync::oneshot::channel();
535        self.shutdown_tx = Some(shutdown_tx);
536
537        let server_handle = tokio::spawn(async move {
538            let graceful = axum::serve(listener, router)
539                .with_graceful_shutdown(async {
540                    shutdown_rx.await.ok();
541                });
542
543            if let Err(e) = graceful.await {
544                error!("Server error: {}", e);
545            }
546        });
547
548        self.server_handle = Some(server_handle);
549        self.running = true;
550
551        info!("HTTP server {} started on {}:{}", self.id, host, self.port);
552        Ok(self.port)
553    }

555    async fn start_https(&mut self, actor_handle: ActorHandle, tls_config: &TlsConfig) -> Result<u16> {
556        // Validate TLS configuration early
557        debug!("Validating TLS configuration for server {}", self.id);
558        validate_tls_config(&tls_config.cert_path, &tls_config.key_path)
559            .map_err(|e| anyhow::anyhow!("TLS validation failed: {}", e))?;
560
561        // Build router
562        let router = self.build_router_safe(actor_handle).map_err(|e| anyhow::anyhow!(e))?;
563        let host = self.config.host.clone().unwrap_or_else(|| "0.0.0.0".to_string());
564        
565        let addr = if self.port == 0 {
566            format!("{}:0", host)
567        } else {
568            format!("{}:{}", host, self.port)
569        };
570
571        // Create TLS configuration
572        debug!("Loading TLS certificates for server {}", self.id);
573        let rustls_config = create_tls_config(&tls_config.cert_path, &tls_config.key_path)
574            .map_err(|e| anyhow::anyhow!("Failed to create TLS config: {}", e))?;
575
576        // Bind to address
577        let listener = tokio::net::TcpListener::bind(&addr).await?;
578        let actual_addr = listener.local_addr()?;
579        self.port = actual_addr.port();
580
581        let (shutdown_tx, shutdown_rx) = tokio::sync::oneshot::channel();
582        self.shutdown_tx = Some(shutdown_tx);
583
584        // Convert to std::net::TcpListener for axum-server
585        let std_listener = listener.into_std()?;
586        
587        // Start HTTPS server using axum-server
588        let server_handle = tokio::spawn(async move {
589            let server = axum_server::from_tcp_rustls(std_listener, rustls_config)
590                .serve(router.into_make_service());
591                
592            tokio::select! {
593                result = server => {
594                    if let Err(e) = result {
595                        error!("HTTPS server error: {}", e);
596                    }
597                }
598                _ = shutdown_rx => {
599                    debug!("HTTPS server received shutdown signal");
600                }
601            }
602        });
603
604        self.server_handle = Some(server_handle);
605        self.running = true;
606
607        info!("HTTPS server {} started on {}:{}", self.id, host, self.port);
608        Ok(self.port)
609    }

611    pub async fn stop(&mut self) -> Result<()> {
612        if !self.running {
613            return Ok(());
614        }
615
616        if let Some(shutdown_tx) = self.shutdown_tx.take() {
617            let _ = shutdown_tx.send(());
618        }
619
620        if let Some(handle) = self.server_handle.take() {
621            handle.abort();
622            
623            let connections_count = self.active_ws_connections.read().await.len();
624            if connections_count > 0 {
625                debug!(
626                    "Closing {} WebSocket connections for server {}",
627                    connections_count, self.id
628                );
629                let mut connections = self.active_ws_connections.write().await;
630                connections.clear();
631            }
632        }
633
634        self.running = false;
635        info!("HTTP server {} on port {} stopped", self.id, self.port);
636
637        tokio::time::sleep(std::time::Duration::from_millis(200)).await;
638        Ok(())
639    }

641    fn build_router_safe(&self, actor_handle: ActorHandle) -> Result<Router, String> {
642        let result = panic::catch_unwind(panic::AssertUnwindSafe(|| {
643            self.build_router(actor_handle)
644        }));
645
646        match result {
647            Ok(router) => Ok(router),
648            Err(panic_payload) => {
649                let panic_msg = if let Some(s) = panic_payload.downcast_ref::<String>() {
650                    s.clone()
651                } else if let Some(s) = panic_payload.downcast_ref::<&str>() {
652                    s.to_string()
653                } else {
654                    "Unknown panic occurred while building router".to_string()
655                };
656
657                let error_msg = if panic_msg.contains("Path segments must not start with `*`") {
658                    let suggestion = "Path segments must not start with `*`. For wildcard capture, use `{*wildcard}` syntax instead.";
659                    format!("Invalid route pattern: {}. {}", panic_msg, suggestion)
660                } else if panic_msg.contains("without_v07_checks") {
661                    format!("Route pattern validation failed: {}. Please update route patterns to use Axum v0.7+ syntax.", panic_msg)
662                } else {
663                    format!("Router building failed: {}", panic_msg)
664                };
665
666                Err(error_msg)
667            }
668        }
669    }

673    pub fn find_middlewares(&self, path: &str) -> Vec<&MiddlewareConfig> {
674        let mut result: Vec<&MiddlewareConfig> = self
675            .middlewares
676            .values()
677            .filter(|middleware| {
678                // Match if the middleware path is a prefix of the request path
679                // or is exactly the request path or is catch-all
680                path.starts_with(&middleware.path) || middleware.path == "/*"
681            })
682            .collect();
683
684        result.sort_by(|a, b| a.priority.cmp(&b.priority));
685        result
686    }

721// Route pattern validation helper
722fn validate_route_pattern(path: &str) -> Result<(), String> {
723    // Basic validation for common Axum v0.7+ pattern issues
724    if path.contains("/*") && !path.contains("/{*") {
725        return Err(format!(
726            "Invalid wildcard pattern '{}'. Use '{{*wildcard}}' instead of '*' for wildcard capture.",
727            path
728        ));
729    }
730
731    // Check for other common pattern issues
732    if path.contains("{") && !path.contains("}") {
733        return Err(format!("Unmatched '{{' in route pattern: {}", path));
734    }
735
736    if path.contains("}") && !path.contains("{") {
737        return Err(format!("Unmatched '}}' in route pattern: {}", path));
738    }
739
740    Ok(())
741}

743// NEW: Enhanced request conversion that includes Axum-extracted path parameters
744async fn convert_request_with_axum_params(
745    req: Request<axum::body::Body>,
746    path_params: Option<Path<HashMap<String, String>>>,
747) -> HttpRequest {
748    let method = req.method().as_str().to_string();
749    let uri = req.uri().to_string();
750    
751    // Extract headers
752    let mut headers = Vec::new();
753    for (name, value) in req.headers() {
754        if let Ok(value_str) = value.to_str() {
755            headers.push((name.to_string(), value_str.to_string()));
756        }
757    }
758    
759    // Add path parameters as special headers so WASM handlers can access them
760    if let Some(Path(params)) = path_params {
761        for (key, value) in params {
762            headers.push((format!("X-Route-Param-{}", key), value.clone()));
763            debug!("Extracted path parameter: {} = {}", key, value);
764        }
765    }
766    
767    // Read body
768    let body = match axum::body::to_bytes(req.into_body(), 100 * 1024 * 1024).await {
769        Ok(bytes) => Some(bytes.to_vec()),
770        Err(e) => {
771            error!("Failed to read request body: {}", e);
772            None
773        }
774    };
775
776    HttpRequest {
777        method,
778        uri,
779        headers,
780        body,
781    }
782}

784// Helper function to convert Theater response to Axum response
785fn convert_response(response: HttpResponse) -> Response<axum::body::Body> {
786    let mut builder = Response::builder().status(response.status);
787
788    // Add headers
789    for (name, value) in response.headers {
790        if let (Ok(header_name), Ok(header_value)) = (
791            HeaderName::try_from(name),
792            HeaderValue::try_from(value),
793        ) {
794            builder = builder.header(header_name, header_value);
795        }
796    }
797
798    // Set body
799    let body = response.body.unwrap_or_default();
800    builder.body(axum::body::Body::from(body)).unwrap_or_default()
801}
