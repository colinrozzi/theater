12/// Load certificates from a PEM file
13pub fn load_certs(path: &str) -> Result<Vec<CertificateDer<'static>>> {
14    let path = Path::new(path);
15    
16    if !path.exists() {
17        return Err(anyhow::anyhow!("Certificate file not found: {}", path.display()));
18    }
19    
20    if !path.is_file() {
21        return Err(anyhow::anyhow!("Certificate path is not a file: {}", path.display()));
22    }
23    
24    debug!("Loading certificates from: {}", path.display());
25    
26    let file = File::open(path)
27        .with_context(|| format!("Failed to open certificate file: {}", path.display()))?;
28    
29    let mut reader = BufReader::new(file);
30    let certs: Result<Vec<_>, _> = certs(&mut reader).collect();
31    let certs = certs
32        .with_context(|| format!("Failed to parse certificates from: {}", path.display()))?;
33    
34    if certs.is_empty() {
35        return Err(anyhow::anyhow!("No certificates found in file: {}", path.display()));
36    }
37    
38    info!("Loaded {} certificate(s) from: {}", certs.len(), path.display());
39    Ok(certs)
40}

42/// Load private key from a PEM file
43pub fn load_private_key(path: &str) -> Result<PrivateKeyDer<'static>> {
44    let path = Path::new(path);
45    
46    if !path.exists() {
47        return Err(anyhow::anyhow!("Private key file not found: {}", path.display()));
48    }
49    
50    if !path.is_file() {
51        return Err(anyhow::anyhow!("Private key path is not a file: {}", path.display()));
52    }
53    
54    debug!("Loading private key from: {}", path.display());
55    
56    let file = File::open(path)
57        .with_context(|| format!("Failed to open private key file: {}", path.display()))?;
58    
59    let mut reader = BufReader::new(file);
60    let key = private_key(&mut reader)
61        .with_context(|| format!("Failed to parse private key from: {}", path.display()))?
62        .ok_or_else(|| anyhow::anyhow!("No private key found in file: {}", path.display()))?;
63    
64    info!("Loaded private key from: {}", path.display());
65    Ok(key)
66}

68/// Create a rustls ServerConfig from certificate and key paths
69pub fn create_tls_config(cert_path: &str, key_path: &str) -> Result<RustlsConfig> {
70    debug!("Creating TLS configuration with cert: {}, key: {}", cert_path, key_path);
71    
72    // Load certificates
73    let cert_chain = load_certs(cert_path)?;
74    
75    // Load private key
76    let private_key = load_private_key(key_path)?;
77    
78    // Create TLS configuration
79    let server_config = ServerConfig::builder()
80        .with_no_client_auth()
81        .with_single_cert(cert_chain, private_key)
82        .with_context(|| "Failed to create TLS configuration")?;
83    
84    // Convert to RustlsConfig for axum-server
85    let rustls_config = RustlsConfig::from_config(Arc::new(server_config));
86    
87    info!("TLS configuration created successfully");
88    Ok(rustls_config)
89}

91/// Validate TLS configuration without creating the full config
92/// Useful for early validation
93pub fn validate_tls_config(cert_path: &str, key_path: &str) -> Result<()> {
94    debug!("Validating TLS configuration paths");
95    
96    // Check if files exist and are readable
97    let cert_path = Path::new(cert_path);
98    let key_path = Path::new(key_path);
99    
100    if !cert_path.exists() {
101        return Err(anyhow::anyhow!("Certificate file not found: {}", cert_path.display()));
102    }
103    
104    if !key_path.exists() {
105        return Err(anyhow::anyhow!("Private key file not found: {}", key_path.display()));
106    }
107    
108    if !cert_path.is_file() {
109        return Err(anyhow::anyhow!("Certificate path is not a file: {}", cert_path.display()));
110    }
111    
112    if !key_path.is_file() {
113        return Err(anyhow::anyhow!("Private key path is not a file: {}", key_path.display()));
114    }
115    
116    // Try to load them to make sure they're valid
117    let _certs = load_certs(cert_path.to_str().unwrap())?;
118    let _key = load_private_key(key_path.to_str().unwrap())?;
119    
120    debug!("TLS configuration validation successful");
121    Ok(())
122}
