1007    pub async fn start(
1008        &mut self,
1009        actor_handle: ActorHandle,
1010        mut shutdown_receiver: ShutdownReceiver,
1011    ) -> Result<()> {
1012        info!("Starting supervisor host");
1013
1014        loop {
1015            tokio::select! {
1016                Some(child_error) = self.channel_rx.recv() => {
1017                    self.process_child_result(actor_handle.clone(), child_error).await?;
1018                }
1019                _ = &mut shutdown_receiver.receiver => {
1020                    info!("Shutdown signal received");
1021                    break;
1022                }
1023            }
1024        }
1025        info!("Supervisor host shut down complete");
1026        Ok(())
1027    }

1029    async fn process_child_result(
1030        &self,
1031        actor_handle: ActorHandle,
1032        actor_result: ActorResult,
1033    ) -> Result<()> {
1034        info!("Processing child result");
1035
1036        match actor_result {
1037            ActorResult::Error(child_error) => {
1038                actor_handle
1039                    .call_function::<(String, WitActorError), ()>(
1040                        "theater:simple/supervisor-handlers.handle-child-error".to_string(),
1041                        (child_error.actor_id.to_string(), child_error.error.into()),
1042                    )
1043                    .await?;
1044            }
1045            ActorResult::Success(child_result) => {
1046                info!("Child result: {:?}", child_result);
1047                actor_handle
1048                    .call_function::<(String, Option<Vec<u8>>), ()>(
1049                        "theater:simple/supervisor-handlers.handle-child-exit".to_string(),
1050                        (
1051                            child_result.actor_id.to_string(),
1052                            child_result.result.into(),
1053                        ),
1054                    )
1055                    .await?;
1056            }
1057            ActorResult::ExternalStop(stop_data) => {
1058                info!("External stop received for actor: {}", stop_data.actor_id);
1059                actor_handle
1060                    .call_function::<(String,), ()>(
1061                        "theater:simple/supervisor-handlers.handle-child-external-stop".to_string(),
1062                        (stop_data.actor_id.to_string(),),
1063                    )
1064                    .await?;
1065            }
1066        }
1067
1068        Ok(())
1069    }
