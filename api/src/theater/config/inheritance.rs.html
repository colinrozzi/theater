4/// Defines how a handler should inherit permissions from its parent
5#[derive(Debug, Clone, Serialize, Deserialize)]
6#[serde(tag = "type", content = "config")]
7pub enum HandlerInheritance<T> {
8    #[serde(rename = "inherit")]
9    Inherit,
10    #[serde(rename = "disallow")]
11    Disallow,
12    #[serde(rename = "restrict")]
13    Restrict(T),
14}
15
16impl<T> Default for HandlerInheritance<T> {
17    fn default() -> Self {
18        Self::Inherit
19    }
20}

96/// Apply inheritance policy to calculate effective permissions
97pub fn apply_inheritance_policy<T: Clone + RestrictWith<T>>(
98    parent_capability: &Option<T>,
99    policy: &HandlerInheritance<T>,
100) -> Option<T> {
101    match policy {
102        HandlerInheritance::Inherit => parent_capability.clone(),
103        HandlerInheritance::Disallow => None,
104        HandlerInheritance::Restrict(restriction) => parent_capability
105            .as_ref()
106            .map(|parent| parent.restrict_with(restriction)),
107    }
108}
