209impl EnvironmentHandlerConfig {
210    pub fn is_variable_allowed(&self, var_name: &str) -> bool {
211        // Check denied list first
212        if let Some(denied) = &self.denied_vars {
213            if denied.contains(&var_name.to_string()) {
214                return false;
215            }
216        }
217
218        // Check allowed list
219        if let Some(allowed) = &self.allowed_vars {
220            return allowed.contains(&var_name.to_string());
221        }
222
223        // Check allowed prefixes
224        if let Some(prefixes) = &self.allowed_prefixes {
225            return prefixes.iter().any(|prefix| var_name.starts_with(prefix));
226        }
227
228        // If no restrictions are configured, allow all except denied
229        self.denied_vars.is_none()
230            || !self
231                .denied_vars
232                .as_ref()
233                .unwrap()
234                .contains(&var_name.to_string())
235    }
236}

276impl ManifestConfig {
277    /// Loads a manifest configuration from a TOML file.
278    ///
279    /// ## Purpose
280    ///
281    /// This method reads a manifest file from disk, parses it as TOML, and constructs
282    /// a ManifestConfig instance. It's the primary way to load actor configurations
283    /// from the filesystem.
284    ///
285    /// ## Parameters
286    ///
287    /// * `path` - Path to the TOML manifest file
288    ///
289    /// ## Returns
290    ///
291    /// * `Ok(ManifestConfig)` - The successfully parsed configuration
292    /// * `Err(anyhow::Error)` - If the file cannot be read or contains invalid TOML
293    ///
294    /// ## Example
295    ///
296    /// ```rust
297    /// use theater::ManifestConfig;
298    /// use std::path::Path;
299    ///
300    /// fn example() -> anyhow::Result<()> {
301    ///     let config = ManifestConfig::from_file(Path::new("manifest.toml"))?;
302    ///     println!("Loaded actor: {}", config.name);
303    ///     Ok(())
304    /// }
305    /// ```
306    pub fn from_file<P: AsRef<std::path::Path>>(path: P) -> anyhow::Result<Self> {
307        let content = std::fs::read_to_string(path)?;
308        let config: ManifestConfig = toml::from_str(&content)?;
309        Ok(config)
310    }
311
312    /// Loads a manifest configuration from a TOML string.
313    ///
314    /// ## Purpose
315    ///
316    /// This method parses a string containing TOML data and constructs a ManifestConfig
317    /// instance. This is useful when the manifest content is available in memory rather
318    /// than in a file.
319    ///
320    /// ## Parameters
321    ///
322    /// * `content` - TOML string containing the manifest configuration
323    ///
324    /// ## Returns
325    ///
326    /// * `Ok(ManifestConfig)` - The successfully parsed configuration
327    /// * `Err(anyhow::Error)` - If the string contains invalid TOML
328    ///
329    /// ## Example
330    ///
331    /// ```rust
332    /// use theater::ManifestConfig;
333    ///
334    /// fn example() -> anyhow::Result<()> {
335    ///     let toml_content = r#"
336    ///         name = "example-actor"
337    ///         component_path = "./example.wasm"
338    ///     "#;
339    ///     
340    ///     let config = ManifestConfig::from_str(toml_content)?;
341    ///     println!("Loaded actor: {}", config.name);
342    ///     Ok(())
343    /// }
344    /// ```
345    pub fn from_str(content: &str) -> anyhow::Result<Self> {
346        tracing::info!("Parsing manifest TOML content: {}", content);
347        
348        let config: ManifestConfig = match toml::from_str(content) {
349            Ok(config) => {
350                tracing::info!("Successfully parsed manifest TOML");
351                config
352            }
353            Err(e) => {
354                tracing::error!("Failed to parse manifest TOML: {}", e);
355                return Err(e.into());
356            }
357        };
358        
359        // Debug logging to trace permission parsing
360        tracing::info!("Parsed manifest permission_policy: {:?}", config.permission_policy);
361        tracing::info!("Parsed manifest file_system inheritance: {:?}", config.permission_policy.file_system);
362        
363        Ok(config)
364    }

366    /// Loads a manifest configuration from a byte vector.
367    ///
368    /// ## Purpose
369    ///
370    /// This method converts a byte vector to a UTF-8 string, parses it as TOML,
371    /// and constructs a ManifestConfig instance. This is useful when the manifest
372    /// content is available as raw bytes, such as when loaded from a content store.
373    ///
374    /// ## Parameters
375    ///
376    /// * `content` - Byte vector containing UTF-8 encoded TOML data
377    ///
378    /// ## Returns
379    ///
380    /// * `Ok(ManifestConfig)` - The successfully parsed configuration
381    /// * `Err(anyhow::Error)` - If the bytes cannot be converted to valid UTF-8 or contain invalid TOML
382    ///
383    /// ## Example
384    ///
385    /// ```rust
386    /// use theater::ManifestConfig;
387    ///
388    /// fn example() -> anyhow::Result<()> {
389    ///     let bytes = vec![/* ... */];
390    ///     let config = ManifestConfig::from_vec(bytes)?;
391    ///     println!("Loaded actor: {}", config.name);
392    ///     Ok(())
393    /// }
394    /// ```
395    pub fn from_vec(content: Vec<u8>) -> anyhow::Result<Self> {
396        let config: ManifestConfig = toml::from_str(&String::from_utf8(content)?)?;
397        Ok(config)
398    }

423    /// Loads the initial state data for the actor.
424    ///
425    /// ## Purpose
426    ///
427    /// This method reads the initial state data from the file specified in the
428    /// `init_state` field, if present. This data is used to initialize the actor
429    /// when it starts.
430    ///
431    /// ## Returns
432    ///
433    /// * `Ok(Some(Vec<u8>))` - The initial state data if specified and successfully loaded
434    /// * `Ok(None)` - If no initial state file is specified
435    /// * `Err(anyhow::Error)` - If the initial state file cannot be read
436    ///
437    /// ## Example
438    ///
439    /// ```rust
440    /// # use theater::ManifestConfig;
441    /// # async fn example(config: &ManifestConfig) -> anyhow::Result<()> {
442    /// if let Some(state_data) = config.load_init_state().await? {
443    ///     println!("Loaded initial state: {} bytes", state_data.len());
444    /// } else {
445    ///     println!("No initial state specified");
446    /// }
447    /// # Ok(())
448    /// # }
449    /// ```
450    pub async fn load_init_state(&self) -> anyhow::Result<Option<Vec<u8>>> {
451        match &self.init_state {
452            Some(reference) => {
453                let data = resolve_reference(reference).await?;
454                Ok(Some(data))
455            }
456            None => Ok(None),
457        }
458    }

460    /// Converts the manifest to a fixed byte representation.
461    ///
462    /// ## Purpose
463    ///
464    /// This method serializes the manifest to a standardized byte representation
465    /// suitable for content-addressed storage. The goal is to ensure that identical
466    /// manifests produce identical byte representations, enabling deduplication.
467    ///
468    /// ## Returns
469    ///
470    /// A byte vector containing the serialized manifest
471    ///
472    /// ## Example
473    ///
474    /// ```rust
475    /// # use theater::ManifestConfig;
476    /// # fn example(config: ManifestConfig) {
477    /// let bytes = config.into_fixed_bytes();
478    /// println!("Serialized manifest: {} bytes", bytes.unwrap().len());
479    /// # }
480    /// ```
481    ///
482    /// ## Implementation Notes
483    ///
484    /// This is intended to store a manifest in the content store in such a way that there is only
485    /// one representation per possible manifest. The current implementation uses TOML serialization,
486    /// but this might be refined in the future to guarantee consistent representations.
487    pub fn into_fixed_bytes(self) -> Result<Vec<u8>, anyhow::Error> {
488        debug!("Serializing manifest config to fixed bytes");
489        debug!("Manifest config: {:?}", self);
490        let serialized = toml::to_string(&self)
491            .map_err(|e| anyhow::anyhow!("Failed to serialize manifest: {}", e))?;
492        debug!("Serialized manifest config: {}", serialized);
493        Ok(serialized.into_bytes())
494    }

500    /// Calculate effective permissions based on parent permissions and this manifest's policy
501    pub fn calculate_effective_permissions(
502        &self,
503        parent_permissions: &HandlerPermission,
504    ) -> HandlerPermission {
505        tracing::info!("Calculating effective permissions from parent: {:?}", parent_permissions);
506        tracing::info!("Using permission policy: {:?}", self.permission_policy);
507        
508        let effective = HandlerPermission::calculate_effective(parent_permissions, &self.permission_policy);
509        
510        tracing::info!("Calculated effective permissions: {:?}", effective);
511        tracing::info!("Effective filesystem permissions: {:?}", effective.file_system);
512        
513        effective
514    }
