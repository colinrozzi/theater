166    pub allowed_vars: Option<Vec<String>>,

168    pub denied_vars: Option<Vec<String>>,
169    /// Whether to allow listing all environment variables (default: false for security)

173    pub allowed_prefixes: Option<Vec<String>>,

176impl EnvironmentHandlerConfig {
177    pub fn is_variable_allowed(&self, var_name: &str) -> bool {
178        // Check denied list first
179        if let Some(denied) = &self.denied_vars {
180            if denied.contains(&var_name.to_string()) {
181                return false;
182            }
183        }
184
185        // Check allowed list
186        if let Some(allowed) = &self.allowed_vars {
187            return allowed.contains(&var_name.to_string());
188        }
189
190        // Check allowed prefixes
191        if let Some(prefixes) = &self.allowed_prefixes {
192            return prefixes.iter().any(|prefix| var_name.starts_with(prefix));
193        }
194
195        // If no restrictions are configured, allow all except denied
196        self.denied_vars.is_none()
197            || !self
198                .denied_vars
199                .as_ref()
200                .unwrap()
201                .contains(&var_name.to_string())
202    }
203}

244    /// Loads a manifest configuration from a TOML file.
245    ///
246    /// ## Purpose
247    ///
248    /// This method reads a manifest file from disk, parses it as TOML, and constructs
249    /// a ManifestConfig instance. It's the primary way to load actor configurations
250    /// from the filesystem.
251    ///
252    /// ## Parameters
253    ///
254    /// * `path` - Path to the TOML manifest file
255    ///
256    /// ## Returns
257    ///
258    /// * `Ok(ManifestConfig)` - The successfully parsed configuration
259    /// * `Err(anyhow::Error)` - If the file cannot be read or contains invalid TOML
260    ///
261    /// ## Example
262    ///
263    /// ```rust
264    /// use theater::ManifestConfig;
265    /// use std::path::Path;
266    ///
267    /// fn example() -> anyhow::Result<()> {
268    ///     let config = ManifestConfig::from_file(Path::new("manifest.toml"))?;
269    ///     println!("Loaded actor: {}", config.name);
270    ///     Ok(())
271    /// }
272    /// ```
273    pub fn from_file<P: AsRef<std::path::Path>>(path: P) -> anyhow::Result<Self> {
274        let content = std::fs::read_to_string(path)?;
275        let config: ManifestConfig = toml::from_str(&content)?;
276        Ok(config)
277    }

279    /// Loads a manifest configuration from a TOML string.
280    ///
281    /// ## Purpose

283    /// This method parses a string containing TOML data and constructs a ManifestConfig
284    /// instance. This is useful when the manifest content is available in memory rather

286    ///
287    /// ## Parameters
288    ///
289    /// * `content` - TOML string containing the manifest configuration
290    ///
291    /// ## Returns
292    ///
293    /// * `Ok(ManifestConfig)` - The successfully parsed configuration
294    /// * `Err(anyhow::Error)` - If the string contains invalid TOML
295    ///
296    /// ## Example
297    ///
298    /// ```rust
299    /// use theater::ManifestConfig;
300    ///
301    /// fn example() -> anyhow::Result<()> {
302    ///     let toml_content = r#"
303    ///         name = "example-actor"
304    ///         component_path = "./example.wasm"
305    ///     "#;
306    ///     
307    ///     let config = ManifestConfig::from_str(toml_content)?;
308    ///     println!("Loaded actor: {}", config.name);
309    ///     Ok(())
310    /// }
311    /// ```
312    pub fn from_str(content: &str) -> anyhow::Result<Self> {
313        let config: ManifestConfig = toml::from_str(content)?;
314        Ok(config)
315    }

319    /// ## Purpose

321    /// This method converts a byte vector to a UTF-8 string, parses it as TOML,

324    ///
325    /// ## Parameters
326    ///
327    /// * `content` - Byte vector containing UTF-8 encoded TOML data
328    ///
329    /// ## Returns
330    ///
331    /// * `Ok(ManifestConfig)` - The successfully parsed configuration
332    /// * `Err(anyhow::Error)` - If the bytes cannot be converted to valid UTF-8 or contain invalid TOML

334    /// ## Example
335    ///
336    /// ```rust
337    /// use theater::ManifestConfig;
338    ///
339    /// fn example() -> anyhow::Result<()> {

374    /// Loads the initial state data for the actor.
375    ///
376    /// ## Purpose
377    ///
378    /// This method reads the initial state data from the file specified in the
379    /// `init_state` field, if present. This data is used to initialize the actor
380    /// when it starts.
381    ///
382    /// ## Returns
383    ///
384    /// * `Ok(Some(Vec<u8>))` - The initial state data if specified and successfully loaded
385    /// * `Ok(None)` - If no initial state file is specified
386    /// * `Err(anyhow::Error)` - If the initial state file cannot be read
387    ///
388    /// ## Example
389    ///
390    /// ```rust
391    /// # use theater::ManifestConfig;
392    /// # async fn example(config: &ManifestConfig) -> anyhow::Result<()> {
393    /// if let Some(state_data) = config.load_init_state().await? {
394    ///     println!("Loaded initial state: {} bytes", state_data.len());
395    /// } else {
396    ///     println!("No initial state specified");
397    /// }
398    /// # Ok(())
399    /// # }
400    /// ```
401    pub async fn load_init_state(&self) -> anyhow::Result<Option<Vec<u8>>> {
402        match &self.init_state {
403            Some(reference) => {
404                let data = resolve_reference(reference).await?;
405                Ok(Some(data))
406            }
407            None => Ok(None),
408        }
409    }

411    /// Converts the manifest to a fixed byte representation.
412    ///
413    /// ## Purpose
414    ///
415    /// This method serializes the manifest to a standardized byte representation
416    /// suitable for content-addressed storage. The goal is to ensure that identical
417    /// manifests produce identical byte representations, enabling deduplication.
418    ///
419    /// ## Returns
420    ///
421    /// A byte vector containing the serialized manifest
422    ///
423    /// ## Example
424    ///
425    /// ```rust
426    /// # use theater::ManifestConfig;
427    /// # fn example(config: ManifestConfig) {
428    /// let bytes = config.into_fixed_bytes();
429    /// println!("Serialized manifest: {} bytes", bytes.unwrap().len());
430    /// # }
431    /// ```
432    ///
433    /// ## Implementation Notes
434    ///
435    /// This is intended to store a manifest in the content store in such a way that there is only
436    /// one representation per possible manifest. The current implementation uses TOML serialization,
437    /// but this might be refined in the future to guarantee consistent representations.
438    pub fn into_fixed_bytes(self) -> Result<Vec<u8>, anyhow::Error> {
439        debug!("Serializing manifest config to fixed bytes");
440        debug!("Manifest config: {:?}", self);
441        let serialized = toml::to_string(&self)
442            .map_err(|e| anyhow::anyhow!("Failed to serialize manifest: {}", e))?;
443        debug!("Serialized manifest config: {}", serialized);
444        Ok(serialized.into_bytes())
445    }
