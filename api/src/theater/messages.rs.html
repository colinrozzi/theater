2/// # Theater Message System
3///
4/// Defines the message types used for communication between different components
5/// of the Theater system, including commands for the runtime, actor messages,
6/// and channel-based communication.
7///
8/// ## Purpose
9///
10/// This module forms the core messaging infrastructure of Theater, defining the
11/// protocol through which components communicate. It includes command messages for
12/// the runtime to manage actors, messages for inter-actor communication, and the
13/// channel system for streaming communication.
14///
15/// ## Example
16///
17/// ```rust
18/// use theater::messages::{TheaterCommand, ActorStatus};
19/// use theater::id::TheaterId;
20/// use tokio::sync::oneshot;
21///
22/// async fn example() {
23///     // Create a command to spawn a new actor
24///     let (tx, rx) = oneshot::channel();
25///     let spawn_cmd = TheaterCommand::SpawnActor {
26///         manifest_path: "actor_manifest.toml".to_string(),
27///         init_bytes: None,
28///         response_tx: tx,
29///         parent_id: None,
30///         supervisor_tx: None,
31///         subscription_tx: None,
32///     };
33///
34///     // Send the command to the runtime...
35///     
36///     // Wait for the response
37///     let actor_id = rx.await.unwrap().unwrap();
38///     
39///     // Create a command to check actor status
40///     let (status_tx, status_rx) = oneshot::channel();
41///     let status_cmd = TheaterCommand::GetActorStatus {
42///         actor_id,
43///         response_tx: status_tx,
44///     };
45///     
46///     // Send the command and wait for the response
47///     let status = status_rx.await.unwrap().unwrap();
48///     assert_eq!(status, ActorStatus::Running);
49/// }
50/// ```
51///
52/// ## Security
53///
54/// Messages in this module often cross security boundaries between actors
55/// and between actors and the runtime. The message types are designed to
56/// ensure that:
57///
58/// - Actors can only communicate with other actors through controlled channels
59/// - Actor state and events are accessed only by authorized parties
60/// - Commands requiring privileges are properly authenticated
61/// - Channel communication preserves isolation between participants
62///
63/// ## Implementation Notes
64///
65/// The messaging system is built on top of Tokio's `mpsc` and `oneshot` channels
66/// to provide asynchronous communication without blocking. Response channels
67/// (`oneshot::Sender`) are used extensively to allow commands to return results
68/// to their callers.

579/// # Channel Identifier
580///
581/// A unique identifier for a communication channel between participants.
582///
583/// ## Purpose
584///
585/// ChannelId provides a stable, unique identifier for communication channels
586/// between actors or between actors and external components. The identifier
587/// is derived from the participants' identities and includes entropy to
588/// ensure uniqueness.
589///
590/// ## Example
591///
592/// ```rust
593/// use theater::messages::{ChannelId, ChannelParticipant};
594/// use theater::id::TheaterId;
595///
596/// // Create participants
597/// let actor_id = TheaterId::generate();
598/// let initiator = ChannelParticipant::Actor(actor_id);
599/// let target = ChannelParticipant::External;
600///
601/// // Generate a channel ID
602/// let channel_id = ChannelId::new(&initiator, &target);
603/// println!("Created channel: {}", channel_id);
604/// ```
605///
606/// ## Security
607///
608/// Channel IDs include cryptographic entropy to prevent guessing, ensuring
609/// that only authorized participants can access a channel.
610///
611/// ## Implementation Notes
612///
613/// The Channel ID is constructed using a combination of:
614/// - Hashes of both participant identities
615/// - Current timestamp
616/// - Random value
617///
618/// This provides strong uniqueness guarantees even with high channel creation rates.
619#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq, Hash)]
620pub struct ChannelId(pub String);
621
622impl std::fmt::Display for ChannelId {
623    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
624        write!(f, "{}", self.0)
625    }
626}
627
628impl ChannelId {
629    /// # Create a new channel ID
630    ///
631    /// Generates a new unique channel ID based on the participants.
632    ///
633    /// ## Parameters
634    ///
635    /// * `initiator` - The participant initiating the channel
636    /// * `target` - The target participant for the channel
637    ///
638    /// ## Returns
639    ///
640    /// A new unique ChannelId
641    pub fn new(initiator: &ChannelParticipant, target: &ChannelParticipant) -> Self {
642        let mut hasher = DefaultHasher::new();
643        let timestamp = chrono::Utc::now().timestamp_millis();
644        let rand_value: u64 = rand::random();
645
646        initiator.hash(&mut hasher);
647        target.hash(&mut hasher);
648        timestamp.hash(&mut hasher);
649        rand_value.hash(&mut hasher);
650
651        let hash = hasher.finish();
652        ChannelId(format!("ch_{:016x}", hash))
653    }
654
655    /// # Get the channel ID as a string
656    ///
657    /// Returns the string representation of the channel ID.
658    ///
659    /// ## Returns
660    ///
661    /// A string slice containing the channel ID
662    pub fn as_str(&self) -> &str {
663        &self.0
664    }
665}

867/// # Actor Message
868///
869/// Represents the different types of messages that can be sent to an actor.
870///
871/// ## Purpose
872///
873/// ActorMessage provides a unified type for all messages that can be sent to
874/// actors, encompassing request-response interactions, one-way messages,
875/// and channel-based communication.
876///
877/// ## Example
878///
879/// ```rust
880/// use theater::messages::{ActorMessage, ActorSend};
881///
882/// // Create a simple message
883/// let message_data = vec![1, 2, 3, 4]; // Some serialized data
884/// let message = ActorMessage::Send(ActorSend {
885///     data: message_data,
886/// });
887///
888/// // This would then be sent to an actor...
889/// ```
890///
891/// ## Security
892///
893/// The runtime validates that senders have permission to send messages
894/// to the target actor before delivery.
895#[derive(Debug)]
896pub enum ActorMessage {
897    /// Request-response interaction
898    Request(ActorRequest),
899    /// One-way message
900    Send(ActorSend),
901    /// Request to open a new channel
902    ChannelOpen(ActorChannelOpen),
903    /// Message on an established channel
904    ChannelMessage(ActorChannelMessage),
905    /// Request to close a channel
906    ChannelClose(ActorChannelClose),
907    /// Notification of a new channel
908    ChannelInitiated(ActorChannelInitiated),
909}

911/// # Actor Status
912///
913/// Represents the current operational status of an actor.
914///
915/// ## Purpose
916///
917/// ActorStatus provides a standardized way to report the current state of an actor,
918/// used by monitoring tools, supervisors, and the runtime to track actor health.
919///
920/// ## Example
921///
922/// ```rust
923/// use theater::messages::ActorStatus;
924///
925/// // Check if an actor is functioning
926/// fn is_actor_healthy(status: &ActorStatus) -> bool {
927///     matches!(status, ActorStatus::Running)
928/// }
929/// ```
930///
931/// ## Implementation Notes
932///
933/// Status transitions are managed by the runtime and triggered by various events:
934/// - System startup or explicit start commands transition to Running
935/// - Clean shutdown requests transition to Stopped
936/// - Errors or crashes transition to Failed
937#[derive(Debug, Clone, Serialize, Deserialize, PartialEq)]
938pub enum ActorStatus {
939    /// Actor is active and processing messages
940    Running,
941    /// Actor has been stopped gracefully
942    Stopped,
943    /// Actor has experienced an error or crash
944    Failed,
945}
