141impl ManifestConfig {
142    pub fn from_file<P: AsRef<std::path::Path>>(path: P) -> anyhow::Result<Self> {
143        let content = std::fs::read_to_string(path)?;
144        let config: ManifestConfig = toml::from_str(&content)?;
145        Ok(config)
146    }
147
148    pub fn from_str(content: &str) -> anyhow::Result<Self> {
149        let config: ManifestConfig = toml::from_str(content)?;
150        Ok(config)
151    }
152
153    pub fn from_vec(content: Vec<u8>) -> anyhow::Result<Self> {
154        let config: ManifestConfig = toml::from_str(&String::from_utf8(content)?)?;
155        Ok(config)
156    }
157
158    pub fn name(&self) -> &str {
159        &self.name
160    }
161
162    pub fn implements_interface(&self, interface_name: &str) -> bool {
163        self.interface.implements == interface_name
164    }
165
166    pub fn interface(&self) -> &str {
167        &self.interface.implements
168    }
169
170    pub fn load_init_state(&self) -> anyhow::Result<Option<Vec<u8>>> {
171        match &self.init_state {
172            Some(path) => {
173                let data = std::fs::read(path)?;
174                Ok(Some(data))
175            }
176            None => Ok(None),
177        }
178    }
179
180    // This is intended to store a manifest in the content store in a such a way that there is only
181    // one representation per possible manifest. I don't know if toml::to_string does this, but I
182    // am going to go forward for now, because this is more a nice to have than need to have at
183    // this point.
184    pub fn into_fixed_bytes(self) -> Vec<u8> {
185        toml::to_string(&self).unwrap().into_bytes()
