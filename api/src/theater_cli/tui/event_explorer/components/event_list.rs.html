12pub fn render_event_list(f: &mut Frame, app: &EventExplorerApp, area: Rect) {
13    let title = format!(
14        " Events ({}/{}) ",
15        app.filtered_events.len(),
16        app.events.len()
17    );
18
19    let block = Block::default()
20        .title(title)
21        .borders(Borders::ALL)
22        .border_style(if app.show_help {
23            Style::default().fg(Color::Gray)
24        } else {
25            Style::default().fg(Color::Blue)
26        });
27
28    // Create list items from filtered events
29    let items: Vec<ListItem> = app
30        .filtered_events
31        .iter()
32        .enumerate()
33        .map(|(i, &event_idx)| {
34            let event = &app.events[event_idx];
35            let is_selected = Some(i) == app.selected_event_index;
36            create_event_list_item(event, is_selected)
37        })
38        .collect();
39
40    let list = List::new(items)
41        .block(block)
42        .highlight_style(
43            Style::default()
44                .fg(Color::Yellow)
45                .add_modifier(Modifier::BOLD),
46        )
47        .highlight_symbol("‚ñ∂ ");
48
49    f.render_stateful_widget(list, area, &mut app.list_state.clone());
50}

52fn create_event_list_item(event: &theater::chain::ChainEvent, is_selected: bool) -> ListItem {
53    let timestamp = format_timestamp(event.timestamp);
54    let event_type = &event.event_type;
55
56    // Determine event level/color based on event type or description
57    let (level_color, level_symbol) = categorize_event(event);
58
59    let mut spans = vec![
60        Span::styled(
61            format!("[{}] ", timestamp),
62            Style::default().fg(Color::Gray),
63        ),
64        Span::styled(level_symbol, Style::default().fg(level_color)),
65        Span::styled(" ", Style::default()),
66        Span::styled(
67            event_type,
68            if is_selected {
69                Style::default()
70                    .fg(Color::Yellow)
71                    .add_modifier(Modifier::BOLD)
72            } else {
73                Style::default().fg(level_color)
74            },
75        ),
76    ];
77
78    // Add description if it's different from event_type and not too long
79    if let Some(description) = &event.description {
80        if description != event_type && !description.is_empty() {
81            let truncated = if description.len() > 40 {
82                format!("{}...", &description[..37])
83            } else {
84                description.clone()
85            };
86
87            spans.push(Span::styled(
88                format!(" - {}", truncated),
89                Style::default().fg(if is_selected {
90                    Color::White
91                } else {
92                    Color::Gray
93                }),
94            ));
95        }
96    }
97
98    ListItem::new(Line::from(spans))
99}

101fn format_timestamp(timestamp: u64) -> String {
102    let dt = Utc
103        .timestamp_opt(timestamp as i64, 0)
104        .single()
105        .unwrap_or_else(|| Utc::now());
106    dt.format("%H:%M:%S").to_string()
107}

109fn categorize_event(event: &theater::chain::ChainEvent) -> (Color, &'static str) {
110    let event_type = &event.event_type;
111    let description = event.description.as_deref().unwrap_or("");
112
113    // Check for error indicators
114    if event_type.contains("error") || description.to_lowercase().contains("error") {
115        return (Color::Red, "‚ùå");
116    }
117
118    // Check for warning indicators
119    if event_type.contains("warn") || description.to_lowercase().contains("warn") {
120        return (Color::Yellow, "‚ö†");
121    }
122
123    // Check for HTTP events
124    if event_type.starts_with("http") {
125        return (Color::Green, "üåê");
126    }
127
128    // Check for runtime events
129    if event_type.starts_with("runtime") {
130        return (Color::Cyan, "‚öô");
131    }
132
133    // Check for WASM events
134    if event_type.starts_with("wasm") {
135        return (Color::Magenta, "üì¶");
136    }
137
138    // Default
139    (Color::White, "‚Ñπ")
140}
