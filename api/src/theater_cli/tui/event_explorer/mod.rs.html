18pub async fn run_explorer(
19    mut app: EventExplorerApp,
20    _ctx: &CommandContext,
21    _server_address: SocketAddr,
22) -> CliResult<()> {
23    debug!("Starting event explorer TUI");
24
25    // Setup terminal
26    enable_raw_mode().map_err(|e| {
27        crate::error::CliError::Internal(anyhow::anyhow!("Failed to enable raw mode: {}", e))
28    })?;
29
30    let mut stdout = io::stdout();
31    execute!(stdout, EnterAlternateScreen).map_err(|e| {
32        crate::error::CliError::Internal(anyhow::anyhow!("Failed to enter alternate screen: {}", e))
33    })?;
34
35    let backend = CrosstermBackend::new(stdout);
36    let mut terminal = Terminal::new(backend).map_err(|e| {
37        crate::error::CliError::Internal(anyhow::anyhow!("Failed to create terminal: {}", e))
38    })?;
39
40    // Hide cursor
41    terminal.hide_cursor().map_err(|e| {
42        crate::error::CliError::Internal(anyhow::anyhow!("Failed to hide cursor: {}", e))
43    })?;
44
45    // Main event loop
46    let result = run_explorer_loop(&mut terminal, &mut app, _ctx, _server_address).await;
47
48    // Cleanup terminal
49    disable_raw_mode().ok();
50    execute!(terminal.backend_mut(), LeaveAlternateScreen).ok();
51    terminal.show_cursor().ok();
52
53    result
54}

56async fn run_explorer_loop(
57    terminal: &mut Terminal<CrosstermBackend<io::Stdout>>,
58    app: &mut EventExplorerApp,
59    _ctx: &CommandContext,
60    _server_address: SocketAddr,
61) -> CliResult<()> {
62    use input::{handle_input, ExplorerAction};
63
64    debug!("Starting event explorer main loop");
65
66    loop {
67        // Render UI
68        terminal
69            .draw(|f| ui::render_explorer_ui(f, app))
70            .map_err(|e| {
71                crate::error::CliError::Internal(anyhow::anyhow!("Failed to draw terminal: {}", e))
72            })?;
73
74        // Handle input
75        match handle_input()? {
76            ExplorerAction::Quit => {
77                debug!("User requested quit");
78                break;
79            }
80            ExplorerAction::NavigateUp => {
81                app.select_previous();
82            }
83            ExplorerAction::NavigateDown => {
84                app.select_next();
85            }
86            ExplorerAction::PageUp => {
87                app.page_up();
88            }
89            ExplorerAction::PageDown => {
90                app.page_down();
91            }
92            ExplorerAction::ScrollDetailUp => {
93                app.scroll_detail_up();
94            }
95            ExplorerAction::ScrollDetailDown => {
96                app.scroll_detail_down();
97            }
98            ExplorerAction::ScrollDetailPageUp => {
99                app.scroll_detail_page_up(10);
100            }
101            ExplorerAction::ScrollDetailPageDown => {
102                app.scroll_detail_page_down(10);
103            }
104            ExplorerAction::ToggleDetailMode => {
105                app.cycle_detail_mode();
106            }
107            ExplorerAction::ShowHelp => {
108                app.toggle_help();
109            }
110            ExplorerAction::TogglePause => {
111                app.toggle_pause();
112            }
113            ExplorerAction::ToggleFollow => {
114                app.toggle_follow();
115            }
116            ExplorerAction::StartSearch => {
117                app.enter_search_mode();
118            }
119            ExplorerAction::ClearSearch => {
120                app.clear_search();
121            }
122            ExplorerAction::StartFilter => {
123                app.enter_filter_mode();
124            }
125            ExplorerAction::None => {}
126            _ => {
127                // Handle other actions in future phases
128                debug!("Unhandled action received");
129            }
130        }
131
132        // Handle live updates if in live mode
133        if app.live_mode && !app.paused {
134            // TODO: In future phases, check for new events from Theater client
135            // For now, this is a placeholder for live event streaming
136        }
137
138        if app.should_quit {
139            debug!("App requested quit");
140            break;
141        }
142
143        // Prevent excessive CPU usage
144        tokio::time::sleep(tokio::time::Duration::from_millis(16)).await; // ~60 FPS
145    }
146
147    debug!("Event explorer loop ended");
148    Ok(())
149}
