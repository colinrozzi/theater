1use anyhow::{anyhow, Result};
2use clap::Parser;
3use console::style;
4use std::net::SocketAddr;
5use std::time::Duration;
6use tracing::debug;
7
8use crate::client::TheaterClient;
9use std::str::FromStr;
10use theater::id::TheaterId;
11
12#[derive(Debug, Parser)]
13pub struct LogsArgs {
14    /// ID of the actor to get logs from
15    #[arg(required = true)]
16    pub actor_id: String,
17
18    /// Address of the theater server
19    #[arg(short, long, default_value = "127.0.0.1:9000")]
20    pub address: SocketAddr,
21
22    /// Follow logs in real-time
23    #[arg(short, long, default_value = "false")]
24    pub follow: bool,
25
26    /// Number of lines to show (0 for all)
27    #[arg(short, long, default_value = "10")]
28    pub lines: usize,
29}
30
31pub fn execute(args: &LogsArgs, _verbose: bool, json: bool) -> Result<()> {
32    debug!("Getting logs for actor: {}", args.actor_id);
33    debug!("Connecting to server at: {}", args.address);
34
35    // Parse the actor ID
36    let actor_id = TheaterId::from_str(&args.actor_id)
37        .map_err(|_| anyhow!("Invalid actor ID: {}", args.actor_id))?;
38
39    // Create runtime and connect to the server
40    let runtime = tokio::runtime::Runtime::new()?;
41
42    runtime.block_on(async {
43        let mut client = TheaterClient::new(args.address);
44
45        // Connect to the server
46        client.connect().await?;
47
48        // Get the actor events (we'll filter for log events)
49        let events = client.get_actor_events(actor_id.clone()).await?;
50
51        // Filter for log events and extract log messages
52        let log_events: Vec<_> = events.iter().filter(|e| e.event_type == "Log").collect();
53
54        // Limit the number of logs if requested
55        let logs_to_show = if args.lines > 0 && log_events.len() > args.lines {
56            &log_events[log_events.len() - args.lines..]
57        } else {
58            &log_events[..]
59        };
60
61        // Output the logs
62        if json {
63            let output = serde_json::json!({
64                "actor_id": actor_id.to_string(),
65                "logs": logs_to_show,
66                "count": logs_to_show.len()
67            });
68            println!("{}", serde_json::to_string_pretty(&output)?);
69        } else {
70            println!(
71                "{} Logs for actor: {}",
72                style("ℹ").blue().bold(),
73                style(actor_id.to_string()).cyan()
74            );
75
76            if logs_to_show.is_empty() {
77                println!("  No logs found.");
78            } else {
79                for log in logs_to_show {
80                    // Log event data should contain a message field
81                    let ref data = log.data;
82                    if let Ok(json_data) = serde_json::from_slice::<serde_json::Value>(data) {
83                        if let Some(message) = json_data.get("message").and_then(|m| m.as_str()) {
84                            println!("[{}] {}", log.timestamp, message);
85                        }
86                    }
87                }
88            }
89        }
90
91        // If follow mode is enabled, subscribe to actor events and print new logs
92        if args.follow {
93            if !json {
94                println!(
95                    "\n{} Following logs in real-time. Press Ctrl+C to exit.",
96                    style("ℹ").blue().bold()
97                );
98            }
99
100            // Subscribe to actor events
101            let sub_id = client.subscribe_to_actor(actor_id.clone()).await?;
102
103            // TODO: Implement subscription handling for real-time logs
104            // This would require a more complex setup with a channel to receive events
105            // For now, we'll just pause execution to simulate following logs
106
107            // Simulating follow mode with a 60-second wait
108            // In a real implementation, this would process events as they come in
109            tokio::time::sleep(Duration::from_secs(60)).await;
110
111            // Unsubscribe when done
112            client.unsubscribe_from_actor(actor_id, sub_id).await?;
113        }
114
115        Ok::<(), anyhow::Error>(())
116    })?;
117
118    Ok(())
119}
