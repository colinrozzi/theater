93/// Run the Theater CLI asynchronously with cancellation support
94pub async fn run(
95    cli: Cli,
96    config: config::Config,
97    shutdown_token: tokio_util::sync::CancellationToken,
98) -> anyhow::Result<()> {
99    // Create output manager
100    let output = output::OutputManager::new(config.output.clone());
101
102    // Create a context that contains shared resources
103    let ctx = CommandContext {
104        config,
105        output,
106        verbose: cli.verbose,
107        json: cli.json,
108        shutdown_token: shutdown_token.clone(),
109    };
110
111    // Execute the command with cancellation support
112    let command_future = async {
113        match &cli.command {
114            Commands::Subscribe(args) => commands::subscribe::execute_async(args, &ctx)
115                .await
116                .map_err(|e| anyhow::Error::from(e)),
117            Commands::Create(args) => commands::create::execute_async(args, &ctx)
118                .await
119                .map_err(|e| anyhow::Error::from(e)),
120            Commands::Build(args) => commands::build::execute_async(args, &ctx)
121                .await
122                .map_err(|e| anyhow::Error::from(e)),
123            Commands::List(args) => commands::list::execute_async(args, &ctx)
124                .await
125                .map_err(|e| anyhow::Error::from(e)),
126            Commands::State(args) => commands::state::execute_async(args, &ctx)
127                .await
128                .map_err(|e| anyhow::Error::from(e)),
129            Commands::Events(args) => commands::events::execute_async(args, &ctx)
130                .await
131                .map_err(|e| anyhow::Error::from(e)),
132            Commands::EventsExplore(args) => commands::events_explore::execute_async(args, &ctx)
133                .await
134                .map_err(|e| anyhow::Error::from(e)),
135            Commands::Inspect(args) => commands::inspect::execute_async(args, &ctx)
136                .await
137                .map_err(|e| anyhow::Error::from(e)),
138            Commands::Start(args) => commands::start::execute_async(args, &ctx)
139                .await
140                .map_err(|e| anyhow::Error::from(e)),
141            Commands::Stop(args) => commands::stop::execute_async(args, &ctx)
142                .await
143                .map_err(|e| anyhow::Error::from(e)),
144            Commands::Message(args) => commands::message::execute_async(args, &ctx)
145                .await
146                .map_err(|e| anyhow::Error::from(e)),
147            Commands::Channel(args) => match &args.command {
148                commands::channel::ChannelCommands::Open(open_args) => {
149                    commands::channel::open::execute_async(open_args, &ctx)
150                        .await
151                        .map_err(|e| anyhow::Error::from(e))
152                }
153            },
154            Commands::ListStored(args) => commands::list_stored::execute_async(args, &ctx)
155                .await
156                .map_err(|e| anyhow::Error::from(e)),
157            Commands::Completion(args) => commands::completion::execute_async(args, &ctx)
158                .await
159                .map_err(|e| anyhow::Error::from(e)),
160            Commands::DynamicCompletion(args) => {
161                commands::dynamic_completion::execute_async(args, &ctx)
162                    .await
163                    .map_err(|e| anyhow::Error::from(e))
164            }
165        }
166    };
167
168    // Race the command execution against cancellation
169    let result = tokio::select! {
170        result = command_future => result,
171        _ = shutdown_token.cancelled() => {
172            return Err(anyhow::anyhow!("Operation cancelled"));
173        }
174    };
175
176    // Handle the result
177    match result {
178        Ok(()) => Ok(()),
179        Err(e) => {
180            // Use our enhanced error handling
181            if let Some(cli_error) = e.downcast_ref::<error::CliError>() {
182                ctx.output.error(&cli_error.user_message())?;
183                if ctx.verbose {
184                    eprintln!("\nDebug info: {:?}", cli_error);
185                }
186            } else {
187                ctx.output.error(&format!("Error: {}", e))?;
188                if ctx.verbose {
189                    eprintln!("\nDebug info: {:?}", e);
190                }
191            }
192            std::process::exit(1);
193        }
194    }
195}

197/// Shared context for command execution
198pub struct CommandContext {
199    pub config: config::Config,
200    pub output: output::OutputManager,
201    pub verbose: bool,
202    pub json: bool,
203    pub shutdown_token: tokio_util::sync::CancellationToken,
204}
205
206impl CommandContext {
207    /// Create a theater client using the configured server address
208    pub fn create_client(&self) -> client::TheaterClient {
209        client::TheaterClient::new(self.config.server.default_address, self.shutdown_token.clone())
210    }
211
212    /// Get the server address from config or override
213    pub fn server_address(
214        &self,
215        override_addr: Option<std::net::SocketAddr>,
216    ) -> std::net::SocketAddr {
217        override_addr.unwrap_or(self.config.server.default_address)
218    }
