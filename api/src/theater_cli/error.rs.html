5/// Main error type for the Theater CLI
6#[derive(Error, Debug)]
7pub enum CliError {
8    /// Connection-related errors
9    #[error("Failed to connect to Theater server at {address}")]
10    ConnectionFailed {
11        address: SocketAddr,
12        #[source]
13        source: anyhow::Error,
14    },
15
16    #[error("Connection lost to Theater server")]
17    ConnectionLost,
18
19    #[error("Connection timeout after {timeout}s")]
20    ConnectionTimeout { timeout: u64 },
21
22    /// Actor-related errors
23    #[error("Actor '{actor_id}' not found")]
24    ActorNotFound { actor_id: String },
25
26    #[error("Actor '{actor_id}' failed to start: {reason}")]
27    ActorStartFailed { actor_id: String, reason: String },
28
29    #[error("Actor '{actor_id}' is not running")]
30    ActorNotRunning { actor_id: String },
31
32    #[error("Actor '{actor_id}' had an error: {reason}")]
33    ActorError { actor_id: String, reason: String },
34
35    #[error("Actor '{actor_id}' was stopped externally: {reason}")]
36    ActorStopped { actor_id: String, reason: String },
37
38    /// Project and build errors
39    #[error("Invalid project directory: {path}")]
40    InvalidProjectDirectory { path: String },
41
42    #[error("Build failed: {output}")]
43    BuildFailed { output: String },
44
45    #[error("Missing required tool: {tool}. Please install it with: {install_command}")]
46    MissingTool {
47        tool: String,
48        install_command: String,
49    },
50
51    /// Manifest and configuration errors
52    #[error("Invalid manifest file: {reason}")]
53    InvalidManifest { reason: String },
54
55    #[error("Configuration error: {reason}")]
56    ConfigError { reason: String },
57
58    /// Template errors
59    #[error("Template '{template}' not found. Available templates: {available}")]
60    TemplateNotFound { template: String, available: String },
61
62    #[error("Template error: {reason}")]
63    TemplateError { reason: String },
64
65    /// I/O and filesystem errors
66    #[error("File operation failed: {operation} on {path}: {source}")]

68        operation: String,
69        path: String,
70        #[source]
71        source: std::io::Error,
72    },
73

75    PermissionDenied { operation: String, path: String },

77    /// Validation errors

79    InvalidInput {
80        field: String,
81        value: String,
82        suggestion: String,
83    },
84
85    #[error("Validation failed: {reason}")]
86    ValidationError { reason: String },
87
88    /// Server and protocol errors
89    #[error("Server error: {message}")]
90    ServerError { message: String },
91
92    #[error("Protocol error: {reason}")]
93    ProtocolError { reason: String },
94
95    #[error("Unexpected response from server: {response}")]
96    UnexpectedResponse { response: String },
97
98    /// Event and monitoring errors
99    #[error("Event stream error: {reason}")]
100    EventStreamError { reason: String },
101
102    #[error("Event filter error: {filter} is invalid")]
103    EventFilterError { filter: String },
104
105    /// Cancellation errors
106    #[error("Operation was cancelled by user")]
107    OperationCancelled,
108
109    /// Generic wrapper for other errors

111    Internal(#[from] anyhow::Error),
112
113    #[error("Serialization error: {0}")]
114    Serialization(#[from] serde_json::Error),
115
116    #[error("I/O error: {0}")]
117    Io(#[from] std::io::Error),
118
119    /// Network and protocol specific errors

121    NetworkError {
122        operation: String,
123        source: Box<dyn std::error::Error + Send + Sync>,

125
126    #[error("Invalid response: {message}")]

131
132    #[error("I/O operation failed: {operation}")]
133    IoError {
134        operation: String,
135        #[source]

140    ParseError { message: String },

142    #[error("Not implemented: {feature}")]
143    NotImplemented { feature: String, message: String },
144
145    #[error("Server Error: {0}")]
146    ManagementError(#[from] ManagementError),
147}

149impl CliError {
150    /// Create a connection failed error

152        Self::ConnectionFailed {
153            address,
154            source: source.into(),
155        }
156    }
157
158    /// Create an actor not found error
159    pub fn actor_not_found(actor_id: impl Into<String>) -> Self {
160        Self::ActorNotFound {
161            actor_id: actor_id.into(),
162        }
163    }
164
165    /// Create a build failed error
166    pub fn build_failed(output: impl Into<String>) -> Self {
167        Self::BuildFailed {
168            output: output.into(),
169        }
170    }
171
172    /// Create an invalid manifest error
173    pub fn invalid_manifest(reason: impl Into<String>) -> Self {
174        Self::InvalidManifest {
175            reason: reason.into(),
176        }
177    }
178
179    /// Create a template not found error

181        Self::TemplateNotFound {
182            template: template.into(),
183            available: available.join(", "),

185    }
186
187    /// Create a file operation failed error

193        Self::FileOperationFailed {
194            operation: operation.into(),
195            path: path.into(),

197        }
198    }
199
200    /// Create an invalid input error with helpful suggestions
201    pub fn invalid_input(

225    pub fn operation_timeout(_operation: impl Into<String>, timeout: u64) -> Self {

229    /// Create a not implemented error

237    /// Create a management error from a Theater server error
238    pub fn management_error(error: ManagementError) -> Self {
239        Self::ManagementError(error)
240    }

242    /// Get a user-friendly error message with potential solutions
243    pub fn user_message(&self) -> String {
244        match self {
245            Self::ConnectionFailed { address, .. } => {
246                format!(
247                    "Could not connect to Theater server at {}.\n\n\
248                    Possible solutions:\n\
249                    • Start a Theater server with: theater server\n\
250                    • Check if the server address is correct\n\
251                    • Verify the server is running and accessible",
252                    address
253                )
254            }
255            Self::ActorNotFound { actor_id } => {
256                format!(
257                    "Actor '{}' was not found.\n\n\
258                    Possible solutions:\n\
259                    • Check the actor ID is correct\n\
260                    • List running actors with: theater list\n\
261                    • Start the actor with: theater start <manifest>",
262                    actor_id
263                )
264            }
265            Self::BuildFailed { output } => {
266                format!(
267                    "Build failed.\n\n\
268                    Build output:\n{}\n\n\
269                    Possible solutions:\n\
270                    • Check for compilation errors in your Rust code\n\
271                    • Ensure all dependencies are available\n\
272                    • Try a clean build with: theater build --clean",
273                    output
274                )
275            }
276            Self::MissingTool {
277                tool,
278                install_command,
279            } => {
280                format!(
281                    "Required tool '{}' is not installed.\n\n\
282                    Install it with:\n  {}",
283                    tool, install_command
284                )
285            }
286            Self::TemplateNotFound {
287                template,
288                available,
289            } => {
290                format!(
291                    "Template '{}' was not found.\n\n\
292                    Available templates: {}\n\n\
293                    Use one of the available templates or check your template configuration.",
294                    template, available
295                )
296            }
297            Self::InvalidInput {
298                field,
299                value,
300                suggestion,
301            } => {
302                format!("Invalid {}: '{}'\n\n{}", field, value, suggestion)
303            }
304            Self::NotImplemented { feature, message } => {
305                format!(
306                    "Feature '{}' is not implemented.\n\n\
307                    {}\n\n\
308                    ",
309                    feature, message
310                )
311            }
312            _ => self.to_string(),
313        }
314    }

316    /// Check if this error suggests the user should retry
317    pub fn is_retryable(&self) -> bool {
318        matches!(
319            self,
320            Self::ConnectionFailed { .. }
321                | Self::ConnectionLost
322                | Self::ConnectionTimeout { .. }
323                | Self::ServerError { .. }
324        )
325    }
326
327    /// Get the error category for metrics/logging
328    pub fn category(&self) -> &'static str {
329        match self {
330            Self::ConnectionFailed { .. }
331            | Self::ConnectionLost
332            | Self::ConnectionTimeout { .. } => "connection",
333            Self::ActorNotFound { .. }
334            | Self::ActorStartFailed { .. }
335            | Self::ActorNotRunning { .. }
336            | Self::ActorError { .. }
337            | Self::ActorStopped { .. } => "actor",
338            Self::InvalidProjectDirectory { .. }
339            | Self::BuildFailed { .. }
340            | Self::MissingTool { .. } => "build",
341            Self::InvalidManifest { .. } | Self::ConfigError { .. } => "config",
342            Self::TemplateNotFound { .. } | Self::TemplateError { .. } => "template",
343            Self::FileOperationFailed { .. } | Self::PermissionDenied { .. } => "filesystem",
344            Self::InvalidInput { .. } | Self::ValidationError { .. } => "validation",
345            Self::ServerError { .. }
346            | Self::ProtocolError { .. }
347            | Self::UnexpectedResponse { .. } => "server",
348            Self::EventStreamError { .. } | Self::EventFilterError { .. } => "events",
349            Self::OperationCancelled => "cancellation",
350            Self::Internal(_) | Self::Serialization(_) | Self::Io(_) | Self::ParseError { .. } => {
351                "internal"
352            }
353            Self::NetworkError { .. } => "network",
354            Self::InvalidResponse { .. } => "response",
355            Self::IoError { .. } => "io",
356            Self::NotImplemented { .. } => "not_implemented",
357            Self::ManagementError(_) => "management",
358        }
359    }

362/// Result type alias for CLI operations
363pub type CliResult<T> = Result<T, CliError>;
364
365/// Extension trait for converting common errors to CliError
366pub trait IntoCliError<T> {
367    fn into_cli_error(self) -> CliResult<T>;
368    fn with_cli_context(self, context: impl FnOnce() -> CliError) -> CliResult<T>;
369}
370
371impl<T, E> IntoCliError<T> for Result<T, E>
372where
373    E: Into<anyhow::Error>,
374{
375    fn into_cli_error(self) -> CliResult<T> {
376        self.map_err(|e| CliError::Internal(e.into()))
377    }
378
379    fn with_cli_context(self, context: impl FnOnce() -> CliError) -> CliResult<T> {
380        self.map_err(|_| context())
381    }
382}
